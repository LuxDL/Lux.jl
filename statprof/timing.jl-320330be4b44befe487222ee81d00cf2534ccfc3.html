36 gc_total_bytes(gc_num::GC_Num) =
37 gc_num.allocd + gc_num.deferred_alloc + gc_num.total_allocd

39 function GC_Diff(new::GC_Num, old::GC_Num)
40 # logic from `src/gc.c:jl_gc_total_bytes`
41 old_allocd = gc_total_bytes(old)
42 new_allocd = gc_total_bytes(new)
43 return GC_Diff(new_allocd - old_allocd,
44 new.malloc - old.malloc,
45 new.realloc - old.realloc,
46 new.poolalloc - old.poolalloc,
47 new.bigalloc - old.bigalloc,
48 new.freecall - old.freecall,
49 new.total_time - old.total_time,
50 new.pause - old.pause,
51 new.full_sweep - old.full_sweep)
52 end

54 function gc_alloc_count(diff::GC_Diff)
55 diff.malloc + diff.realloc + diff.poolalloc + diff.bigalloc
56 end

91 function prettyprint_getunits(value, numunits, factor)
92 if value == 0 || value == 1
93 return (value, 1)
94 end
95 unit = ceil(Int, log(value) / log(factor))
96 unit = min(numunits, unit)
97 number = value/factor^(unit-1)
98 return number, unit
99 end

118 function time_print(elapsedtime, bytes=0, gctime=0, allocs=0, compile_time=0, newline=false)
119 timestr = Ryu.writefixed(Float64(elapsedtime/1e9), 6)
120 str = sprint() do io
121 print(io, length(timestr) < 10 ? (" "^(10 - length(timestr))) : "")
122 print(io, timestr, " seconds")
123 parens = bytes != 0 || allocs != 0 || gctime > 0 || compile_time > 0
124 parens && print(io, " (")
125 if bytes != 0 || allocs != 0
126 allocs, ma = prettyprint_getunits(allocs, length(_cnt_units), Int64(1000))
127 if ma == 1
128 print(io, Int(allocs), _cnt_units[ma], allocs==1 ? " allocation: " : " allocations: ")
129 else
130 print(io, Ryu.writefixed(Float64(allocs), 2), _cnt_units[ma], " allocations: ")
131 end
132 print(io, format_bytes(bytes))
133 end
134 if gctime > 0
135 if bytes != 0 || allocs != 0
136 print(io, ", ")
137 end
138 print(io, Ryu.writefixed(Float64(100*gctime/elapsedtime), 2), "% gc time")
139 end
140 if compile_time > 0
141 if bytes != 0 || allocs != 0 || gctime > 0
142 print(io, ", ")
143 end
144 print(io, Ryu.writefixed(Float64(100*compile_time/elapsedtime), 2), "% compilation time")
145 end
146 parens && print(io, ")")
147 end
148 newline ? println(str) : print(str)
149 nothing
150 end

152 function timev_print(elapsedtime, diff::GC_Diff, compile_time)
153 allocs = gc_alloc_count(diff)
154 time_print(elapsedtime, diff.allocd, diff.total_time, allocs, compile_time, true)
155 print("elapsed time (ns): $elapsedtime\n")
156 padded_nonzero_print(diff.total_time, "gc time (ns)")
157 padded_nonzero_print(diff.allocd, "bytes allocated")
158 padded_nonzero_print(diff.poolalloc, "pool allocs")
159 padded_nonzero_print(diff.bigalloc, "non-pool GC allocs")
160 padded_nonzero_print(diff.malloc, "malloc() calls")
161 padded_nonzero_print(diff.realloc, "realloc() calls")
162 padded_nonzero_print(diff.freecall, "free() calls")
163 padded_nonzero_print(diff.pause, "GC pauses")
164 padded_nonzero_print(diff.full_sweep, "full collections")
165 end

177 """
178 @time
179
180 A macro to execute an expression, printing the time it took to execute, the number of
181 allocations, and the total number of bytes its execution caused to be allocated, before
182 returning the value of the expression. Any time spent garbage collecting (gc) or
183 compiling is shown as a percentage.
184
185 In some cases the system will look inside the `@time` expression and compile some of the
186 called code before execution of the top-level expression begins. When that happens, some
187 compilation time will not be counted. To include this time you can run `@time @eval ...`.
188
189 See also [`@timev`](@ref), [`@timed`](@ref), [`@elapsed`](@ref), and
190 [`@allocated`](@ref).
191
192 !!! note
193 For more serious benchmarking, consider the `@btime` macro from the BenchmarkTools.jl
194 package which among other things evaluates the function multiple times in order to
195 reduce noise.
196
197 ```julia-repl
198 julia> x = rand(10,10);
199
200 julia> @time x * x;
201 0.606588 seconds (2.19 M allocations: 116.555 MiB, 3.75% gc time, 99.94% compilation time)
202
203 julia> @time x * x;
204 0.000009 seconds (1 allocation: 896 bytes)
205
206 julia> @time begin
207 sleep(0.3)
208 1+1
209 end
210 0.301395 seconds (8 allocations: 336 bytes)
211 2
212 ```
213 """
214 macro time(ex)
215 quote
216 while false; end # compiler heuristic: compile this block (alter this if the heuristic changes)
217 local stats = gc_num()
218 local elapsedtime = time_ns()
219 local compile_elapsedtime = cumulative_compile_time_ns_before()
220 local val = @__tryfinally($(esc(ex)),
221 (elapsedtime = time_ns() - elapsedtime;
222 compile_elapsedtime = cumulative_compile_time_ns_after() - compile_elapsedtime)
223 )
224 local diff = GC_Diff(gc_num(), stats)
225 time_print(elapsedtime, diff.allocd, diff.total_time, gc_alloc_count(diff), compile_elapsedtime, true)
226 val
227 end
