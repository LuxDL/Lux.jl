833 # this is not just an optimization.

883 # reached a fixpoint or iterator failed/gave invalid answer

885 # ... but cannot terminate

888 return Any[Bottom], nothing

890 # iterator may have terminated prior to this loop, but not during it
891 valtype = Bottom
892 end
893 end
894 break
895 end
896 valtype = tmerge(valtype, nounion.parameters[1])
897 statetype = tmerge(statetype, nounion.parameters[2])
898 stateordonet = abstract_call_known(interp, iteratef, nothing, Any[Const(iteratef), itertype, statetype], sv).rt
899 stateordonet = widenconst(stateordonet)
900 end
901 if valtype !== Union{}
902 push!(ret, Vararg{valtype})
903 end
904 return ret, nothing
905 end
