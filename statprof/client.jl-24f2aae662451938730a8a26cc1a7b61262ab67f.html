157 function _parse_input_line_core(s::String, filename::String)
158 ex = Meta.parseall(s, filename=filename)
159 if ex isa Expr && ex.head === :toplevel
160 if isempty(ex.args)
161 return nothing
162 end
163 last = ex.args[end]
164 if last isa Expr && (last.head === :error || last.head === :incomplete)
165 # if a parse error happens in the middle of a multi-line input
166 # return only the error, so that none of the input is evaluated.
167 return last
168 end
169 end
170 return ex
171 end

190 function incomplete_tag(ex::Expr)
191 Meta.isexpr(ex, :incomplete) || return :none
192 msg = ex.args[1]
193 occursin("string", msg) && return :string
194 occursin("comment", msg) && return :comment
195 occursin("requires end", msg) && return :block
