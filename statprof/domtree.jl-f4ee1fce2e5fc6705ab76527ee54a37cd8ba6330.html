3 # This file implements the Semi-NCA (SNCA) dominator tree construction
4 # described in Georgiadis' PhD thesis [LG05], which itself is a simplification
5 # of the Simple Lenguare-Tarjan (SLT) algorithm [LG79]. This algorithm matches
6 # the algorithm choice in LLVM and seems to be a sweet spot in implementation
7 # simplicity and efficiency.
8 #
9 # This file also implements an extension of SNCA that supports updating the
10 # dominator tree with insertion and deletion of edges in the control flow
11 # graph, described in [GI16] as Dynamic SNCA. DSNCA was chosen over DBS, a
12 # different algorithm which achieves the best overall performance in [GI16],
13 # because it is simpler to understand and implement, performs well with edge
14 # deletions, and is of similar performance overall.
15 #
16 # SNCA works by first computing semidominators, then computing immediate
17 # dominators from them. The semidominator of a node is the node with minimum
18 # preorder number such that there is a semidominator path from it to the node.
19 # A semidominator path is a path in which the preorder numbers of all nodes not
20 # at the endpoints are greater than the preorder number of the last node.
21 # Intuitively, the semidominator approximates the immediate dominator of a node
22 # by taking the path (in the CFG) that gets as close to the root as possible
23 # while avoiding ancestors of the node in the DFS tree.
24 #
25 # In computing the semidominators, SNCA performs "path compression" whenever a
26 # node has a nontrivial semidominator (i.e. a semidominator that is not just
27 # its parent in the DFS tree). Path compression propagates the "label" of a
28 # node, which represents a possible semidominator with associated semidominator
29 # path passing through that node.
30 #
31 # For example, path compression will be performed for the following CFG, where
32 # the edge not in the DFS tree is marked with asterisks. Note that nodes are
33 # labeled with their preorder numbers, and all edges point downward.
34 #
35 # 1
36 # |\
37 # | \
38 # | 4
39 # | |
40 # 2 5
41 # | |
42 # | 6
43 # | *
44 # |*
45 # 3
46 #
47 # There is a nontrivial semidominator path from 1 to 3, passing through 4, 5,
48 # and 6. Stepping through the whole algorithm on paper with an example like
49 # this is very helpful for understanding how it works.
50 #
51 # DSNCA runs the whole algorithm from scratch if the DFS tree is invalidated by
52 # the insertion or deletion, but otherwise recomputes a subset of the
53 # semidominators (all immediate dominators then need to be recomputed).
54 #
55 # [LG05] Linear-Time Algorithms for Dominators and Related Problems
56 # Loukas Georgiadis, Princeton University, November 2005, pp. 21-23:
57 # ftp://ftp.cs.princeton.edu/reports/2005/737.pdf
58 #
59 # [LT79] A fast algorithm for finding dominators in a flowgraph
60 # Thomas Lengauer, Robert Endre Tarjan, July 1979, ACM TOPLAS 1-1
61 # http://www.dtic.mil/dtic/tr/fulltext/u2/a054144.pdf
62 #
63 # [GI16] An Experimental Study of Dynamic Dominators
64 # Loukas Georgiadis, Giuseppe F. Italiano, Luigi Laura, Federico
65 # Santaroni, April 2016
66 # https://arxiv.org/abs/1604.02711

246 """
247 The main Semi-NCA algorithm. Matches Figure 2.8 in [LG05]. Note that the
248 pseudocode in [LG05] is not entirely accurate. The best way to understand
249 what's happening is to read [LT79], then the description of SLT in [LG05]
250 (warning: inconsistent notation), then the description of Semi-NCA.
251 """
252 function SNCA!(domtree::DomTree, blocks::Vector{BasicBlock}, max_pre::PreNumber)
253 D = domtree.dfs_tree
254 state = domtree.snca_state
255 # There may be more blocks than are reachable in the DFS / dominator tree
256 n_blocks = length(blocks)
257 n_nodes = length(D)
258
259 # `label` is initialized to the identity mapping (though the paper doesn't
260 # make that clear). The rationale for this is Lemma 2.4 in [LG05] (i.e.
261 # Theorem 4 in [LT79]). Note however, that we don't ever look at `semi`
262 # until it is fully initialized, so we could leave it uninitialized here if
263 # we wanted to.
264 resize!(state, n_nodes)
265 for w in 1:max_pre
266 # Only reset semidominators for nodes we want to recompute
267 state[w] = SNCAData(typemax(PreNumber), w)
268 end
269
270 # If we are only recomputing some of the semidominators, the remaining
271 # labels should be reset, because they may have become inapplicable to the
272 # node/semidominator we are currently processing/recomputing. They can
273 # become inapplicable because of path compressions that were triggered by
274 # nodes that should only be processed after the current one (but were
275 # processed the last time `SNCA!` was run).
276 #
277 # So, for every node that is not being reprocessed, we reset its label to
278 # its semidominator, which is the value that its label assumes once its
279 # semidominator is computed. If this was too conservative, i.e. if the
280 # label would have been updated before we process the current node in a
281 # situation where all semidominators were recomputed, then path compression
282 # will produce the correct label.
283 for w in max_pre+1:n_nodes
284 semi = state[w].semi
285 state[w] = SNCAData(semi, semi)
286 end
287
288 # Calculate semidominators, but only for blocks with preorder number up to
289 # max_pre
290 ancestors = copy(D.to_parent_pre)
291 for w::PreNumber in reverse(2:max_pre)
292 # LLVM initializes this to the parent, the paper initializes this to
293 # `w`, but it doesn't really matter (the parent is a predecessor, so at
294 # worst we'll discover it below). Save a memory reference here.
295 semi_w = typemax(PreNumber)
296 last_linked = PreNumber(w + 1)
297 for v blocks[D.from_pre[w]].preds
298 # For the purpose of the domtree, ignore virtual predecessors into
299 # catch blocks.
300 v == 0 && continue
301
302 v_pre = D.to_pre[v]
303
304 # Ignore unreachable predecessors
305 v_pre == 0 && continue
306
307 # N.B.: This conditional is missing from the pseudocode in figure
308 # 2.8 of [LG05]. It corresponds to the `ancestor[v] != 0` check in
309 # the `eval` implementation in figure 2.6
310 if v_pre >= last_linked
311 # `v` has already been processed, so perform path compression
312
313 # For performance, if the number of ancestors is small avoid
314 # the extra allocation of the worklist.
315 if length(ancestors) <= 32
316 snca_compress!(state, ancestors, v_pre, last_linked)
317 else
318 snca_compress_worklist!(state, ancestors, v_pre, last_linked)
319 end
320 end
321
322 # The (preorder number of the) semidominator of a block is the
323 # minimum over the labels of its predecessors
324 semi_w = min(semi_w, state[v_pre].label)
325 end
326 state[w] = SNCAData(semi_w, semi_w)
327 end
328
329 # Compute immediate dominators, which for a node must be the nearest common
330 # ancestor in the (immediate) dominator tree between its semidominator and
331 # its parent (see Lemma 2.6 in [LG05]).
332 idoms_pre = copy(D.to_parent_pre)
333 for v in 2:n_nodes
334 idom = idoms_pre[v]
335 vsemi = state[v].semi
336 while idom > vsemi
337 idom = idoms_pre[idom]
338 end
339 idoms_pre[v] = idom
340 end
341
342 # Express idoms in BB indexing
343 resize!(domtree.idoms_bb, n_blocks)
344 for i::BBNumber in 1:n_blocks
345 if i == 1 || D.to_pre[i] == 0
346 domtree.idoms_bb[i] = 0
347 else
348 domtree.idoms_bb[i] = D.from_pre[idoms_pre[D.to_pre[i]]]
349 end
350 end
351 end

353 """
354 Matches the snca_compress algorithm in Figure 2.8 of [LG05], with the
355 modification suggested in the paper to use `last_linked` to determine whether
356 an ancestor has been processed rather than storing `0` in the ancestor array.
357 """
358 function snca_compress!(state::Vector{SNCAData}, ancestors::Vector{PreNumber},
359 v::PreNumber, last_linked::PreNumber)
360 u = ancestors[v]
361 @assert u < v
362 if u >= last_linked
363 snca_compress!(state, ancestors, u, last_linked)
364 if state[u].label < state[v].label
365 state[v] = SNCAData(state[v].semi, state[u].label)
366 end
367 ancestors[v] = ancestors[u]
368 end
369 nothing
370 end

372 function snca_compress_worklist!(
373 state::Vector{SNCAData}, ancestors::Vector{PreNumber},
374 v::PreNumber, last_linked::PreNumber)
375 # TODO: There is a smarter way to do this
376 u = ancestors[v]
377 worklist = Tuple{PreNumber, PreNumber}[(u,v)]
378 @assert u < v
379 while !isempty(worklist)
380 u, v = last(worklist)
381 if u >= last_linked
382 if ancestors[u] >= last_linked
383 push!(worklist, (ancestors[u], u))
384 continue
385 end
386 if state[u].label < state[v].label
387 state[v] = SNCAData(state[v].semi, state[u].label)
388 end
389 ancestors[v] = ancestors[u]
390 end
391 pop!(worklist)
392 end

395 "Given updated blocks, update the given dominator tree with an inserted edge."
396 function domtree_insert_edge!(domtree::DomTree, blocks::Vector{BasicBlock},
397 from::BBNumber, to::BBNumber)
398 # `from` is unreachable, so `from` and `to` aren't in domtree
399 if bb_unreachable(domtree, from)
400 return domtree
401 end
402
403 # Implements Section 3.1 of [GI16]
404 dt = domtree.dfs_tree
405 from_pre = dt.to_pre[from]
406 to_pre = dt.to_pre[to]
407 from_post = dt.to_post[from]
408 to_post = dt.to_post[to]
409 if to_pre == 0 || (from_pre < to_pre && from_post < to_post)
410 # The DFS tree is invalidated by the edge insertion, so run from
411 # scratch
412 update_domtree!(blocks, domtree, true, 0)
413 else
414 # DFS tree is still valid, so update only affected nodes
415 update_domtree!(blocks, domtree, false, to_pre)
416 end
417
418 return domtree
419 end

421 "Given updated blocks, update the given dominator tree with a deleted edge."
422 function domtree_delete_edge!(domtree::DomTree, blocks::Vector{BasicBlock},
423 from::BBNumber, to::BBNumber)
424 # `from` is unreachable, so `from` and `to` aren't in domtree
425 if bb_unreachable(domtree, from)
426 return domtree
427 end
428
429 # Implements Section 3.1 of [GI16]
430 if is_parent(domtree.dfs_tree, from, to)
431 # The `from` block is the parent of the `to` block in the DFS tree, so
432 # deleting the edge invalidates the DFS tree, so start from scratch
433 update_domtree!(blocks, domtree, true, 0)
434 elseif on_semidominator_path(domtree, from, to)
435 # Recompute semidominators for blocks with preorder number up to that
436 # of `to` block. Semidominators for blocks with preorder number greater
437 # than that of `to` aren't affected because no semidominator path to
438 # the block can pass through the `to` block (the preorder number of
439 # `to` would be lower than those of these blocks, and `to` is not their
440 # parent in the DFS tree).
441 to_pre = domtree.dfs_tree.to_pre[to]
442 update_domtree!(blocks, domtree, false, to_pre)
443 end
444 # Otherwise, dominator tree is not affected
445
446 return domtree
447 end

456 """
457 Check if x is on some semidominator path from the semidominator of y to y,
458 assuming there is an edge from x to y.
459 """
460 function on_semidominator_path(domtree::DomTree, x::BBNumber, y::BBNumber)
461 x_pre = domtree.dfs_tree.to_pre[x]
462 y_pre = domtree.dfs_tree.to_pre[y]
463
464 semi_y = domtree.snca_state[y_pre].semi
465 current_block = x_pre
466
467 # Follow the semidominators of `x` up the DFS tree to see if we ever reach
468 # the semidominator of `y`. If so, `x` is on a semidominator path between
469 # `y` and its semidominator. We can stop if the preorder number of the
470 # semidominators becomes less than that of the semidominator of `y`,
471 # because it can only decrease further.
472 while current_block >= semi_y
473 if semi_y == current_block
474 return true
475 end
476 current_block = domtree.snca_state[current_block].semi
477 end
478 return false
479 end

543 """
544 Checks if bb1 dominates bb2.
545 bb1 and bb2 are indexes into the CFG blocks.
546 bb1 dominates bb2 if the only way to enter bb2 is via bb1.
547 (Other blocks may be in between, e.g bb1->bbX->bb2).
548 """
549 function dominates(domtree::DomTree, bb1::BBNumber, bb2::BBNumber)
550 bb1 == bb2 && return true
551 target_level = domtree.nodes[bb1].level
552 source_level = domtree.nodes[bb2].level
553 source_level < target_level && return false
554 for _ in (source_level - 1):-1:target_level
555 bb2 = domtree.idoms_bb[bb2]
556 end
557 return bb1 == bb2
558 end

584 function naive_idoms(blocks::Vector{BasicBlock})
585 nblocks = length(blocks)
586 # The extra +1 helps us detect unreachable blocks below
587 dom_all = BitSet(1:nblocks+1)
588 dominators = BitSet[n == 1 ? BitSet(1) : copy(dom_all) for n = 1:nblocks]
589 changed = true
590 while changed
591 changed = false
592 for n = 2:nblocks
593 if isempty(blocks[n].preds)
594 continue
595 end
596 firstp, rest = Iterators.peel(Iterators.filter(p->p != 0, blocks[n].preds))::NTuple{2,Any}
597 new_doms = copy(dominators[firstp])
598 for p in rest
599 intersect!(new_doms, dominators[p])
600 end
601 push!(new_doms, n)
602 changed = changed || (new_doms != dominators[n])
603 dominators[n] = new_doms
604 end
605 end
606 # Compute idoms
607 idoms = fill(0, nblocks)
608 for i = 2:nblocks
609 if dominators[i] == dom_all
610 idoms[i] = 0
611 continue
612 end
613 doms = collect(dominators[i])
614 for dom in doms
615 i == dom && continue
616 hasany = false
617 for p in doms
618 if p !== i && p !== dom && dom in dominators[p]
619 hasany = true; break
620 end
621 end
622 hasany && continue
623 idoms[i] = dom
624 end
625 end
