1 # global state management
2 #
3 # The functionality in this file serves to create a coherent environment to perform
4 # computations in, with support for Julia constructs like tasks (and executing those on
5 # multiple threads), using a GPU of your choice (with the ability to reset that device, or
6 # use different devices on different threads).
7 #
8 # this is complicated by the fact that CUDA uses thread-bound contexts, while we want to be
9 # able to switch between tasks executing on the same thread. the current approach involves
10 # prefixing every CUDA API call with code that ensures the CUDA thread-bound state is
11 # identical to Julia's task-local one.

128
129 Bind the current host thread to the context `ctx`. Returns the previously-bound context. If
130 used with do-block syntax, the change is only temporary.
131
132 Note that the contexts used with this call should be previously acquired by calling
133 [`context`](@ref), and not arbitrary contexts created by calling the `CuContext`
134 constructor.
135 """
136 function context!(ctx::CuContext)
137 # switch contexts
138 # NOTE: if we actually need to switch contexts, we eagerly activate it so that we can
139 # query its device (we normally only do so lazily in `prepare_cuda_state`)
140 state = task_local_state()
141 if state === nothing
142 old_ctx = nothing
143 activate(ctx)
144 dev = current_device()
145 task_local_state!(dev, ctx)
146 else
147 old_ctx = state.context
148 if old_ctx != ctx
149 activate(ctx)
150 dev = current_device()
151 state.device = dev
152 state.context = ctx
153 end
154 end
155
156 return old_ctx
157 end

205 function context(dev::CuDevice)
206 devidx = deviceid(dev)+1
207
208 # querying the primary context for a device is expensive (~100ns), so cache it
209 ctx = device_context(devidx)
210 if ctx !== nothing
211 return ctx
212 end
213
214 if capability(dev) < v"3.5"
215 @warn("""Your $(name(dev)) GPU does not meet the minimal required compute capability ($(capability(dev)) < 3.5).
216 Some functionality might be unavailable.""",
217 maxlog=1, _id=devidx)
218 end
219
220 # configure the primary context
221 pctx = CuPrimaryContext(dev)
222 ctx = CuContext(pctx)
223 device_context!(devidx, ctx)
224 return ctx
225 end

242 function device!(dev::CuDevice, flags=nothing)
243 # configure the primary context flags
244 if flags !== nothing
245 devidx = deviceid(dev)+1
246 if device_context(devidx) !== nothing
247 error("Cannot set flags for an active device. Do so before calling any CUDA function, or reset the device first.")
248 end
249 cuDevicePrimaryCtxSetFlags(dev, flags)
250 end
251
252 # make this device the new default
253 default_device[] = dev
254
255 # switch contexts
256 ctx = context(dev)
257 state = task_local_state()
258 if state === nothing
259 task_local_state!(dev)
260 else
261 state.device = dev
262 state.context = ctx
263 end
264 activate(ctx)
265
266 dev
267 end

277 """
278 device_reset!(dev::CuDevice=device())
279
280 Reset the CUDA state associated with a device. This call with release the underlying
281 context, at which point any objects allocated in that context will be invalidated.
282 """
283 function device_reset!(dev::CuDevice=device())
284 if !can_reset_device()
285 @error "Due to a bug in CUDA, resetting the device is not possible on CUDA 11.2 when using the stream-ordered memory allocator."
286 return
287 end
288
289 # unconditionally reset the primary context (don't just release it),
290 # as there might be users outside of CUDA.jl
291 pctx = CuPrimaryContext(dev)
292 unsafe_reset!(pctx)
293
294 # wipe the device-specific state
295 devidx = deviceid(dev)+1
296 device_context!(devidx, nothing)
297
298 return
299 end

396 """
397 PerDevice{T}()
398
399 A helper struct for maintaining per-device state that's lazily initialized and automatically
400 invalidated when the device is reset. Use `get!(per_device, dev) do ... end` to initialize
401 and fetch a value.
402
403 Mutating or deleting state is not supported. If this is required, use a boxed value, like
404 a `Ref` or a `Threads.Atomic`.
405
406 Furthermore, even though the initialization of this helper, fetching its value for a
407 given device, and clearing it when the device is reset are all performed in a thread-safe
408 manner, you should still take care about thread-safety when using the contained value.
409 For example, if you need to update the value, use atomics; if it's a complex structure like
410 an array or a dictionary, use additional locks.
411 """
412 struct PerDevice{T}
413 lock::ReentrantLock
414 values::LazyInitialized{Vector{Union{Nothing,Tuple{CuContext,T}}}}
415 end
