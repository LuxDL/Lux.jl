643 ptr = convert(Ptr{Cvoid}, pointer(a))

647
648 # derived arrays should always pin the parent memory range, because we may end up copying
649 # from or to that parent range (containing the derived range), and partially-pinned ranges
650 # are not supported:
651 #
652 # > Memory regions requested must be either entirely registered with CUDA, or in the case
653 # > of host pageable transfers, not registered at all. Memory regions spanning over
654 # > allocations that are both registered and not registered with CUDA are not supported and
655 # > will return CUDA_ERROR_INVALID_VALUE.
656 __pin(a::Union{SubArray, Base.ReinterpretArray, Base.ReshapedArray}) = __pin(parent(a))
657
658 # refcount the pinning per context, since we can only pin a memory range once
659 const __pin_lock = ReentrantLock()
660 const __pins = Dict{Tuple{CuContext,Ptr{Cvoid}}, HostBuffer}()
661 const __pin_count = Dict{Tuple{CuContext,Ptr{Cvoid}}, Int}()
662 function __pin(ptr::Ptr{Nothing}, sz::Int)
663 ctx = context()
664 key = (ctx,ptr)
665
666 @lock __pin_lock begin
667 pin_count = if haskey(__pin_count, key)
668 __pin_count[key] += 1
669 else
670 __pin_count[key] = 1
671 end
672 @assert pin_count >= 1 # should have been caught by double-unpin check in __unpin
673
674 if pin_count == 1
675 buf = Mem.register(Mem.Host, ptr, sz)
676 __pins[key] = buf
677 elseif Base.JLOptions().debug_level >= 2
678 # make sure we're pinning the exact same range
679 @assert haskey(__pins, key) "Cannot find buffer for $ptr with pin count $pin_count."
680 buf = __pins[key]
681 @assert sz == sizeof(buf) "Mismatch between pin request of $ptr: $sz vs. $(sizeof(buf))."
682 end
683 end
684
685 return
686 end
687 function __unpin(ptr::Ptr{Nothing}, ctx::CuContext)
688 key = (ctx,ptr)
689
690 @spinlock __pin_lock begin
691 @assert haskey(__pin_count, key) "Cannot unpin unmanaged pointer $ptr."
692 pin_count = __pin_count[key] -= 1
693 @assert pin_count >= 0 "Double unpin for $ptr"
694
695 if pin_count == 0
696 buf = @inbounds __pins[key]
697 context!(ctx; skip_destroyed=true) do
698 Mem.unregister(buf)
699 end
700 delete!(__pins, key)
701 end
702 end
703
704 return
705 end

707 ## p2p handling
708
709 # matrix of set-up peer accesses:
710 # - -1: unsupported
711 # - 0: not set-up yet
712 # - 1: supported
713 const peer_access = Ref{Matrix{Int}}()
714 function maybe_enable_peer_access(src::CuDevice, dst::CuDevice)
715 global peer_access
716
717 src_idx = deviceid(src)+1
718 dst_idx = deviceid(dst)+1
719
720 if !isassigned(peer_access)
721 peer_access[] = Base.zeros(Int8, ndevices(), ndevices())
722 end
723
724 # we need to take care only to enable P2P access when it is supported,
725 # as well as not to call this function multiple times, to avoid errors.
726 if peer_access[][src_idx, dst_idx] == 0
727 if can_access_peer(src, dst)
728 device!(src) do
729 try
730 enable_peer_access(context(dst))
731 if memory_pools_supported(src)
732 src_pool = default_memory_pool(src)
733 access!(src_pool, dst, CUDA.ACCESS_FLAGS_PROT_READWRITE)
734 end
735 peer_access[][src_idx, dst_idx] = 1
736 catch err
737 @warn "Enabling peer-to-peer access between $src and $dst failed; please file an issue." exception=(err,catch_backtrace())
738 peer_access[][src_idx, dst_idx] = -1
739 end
740 end
741 else
742 peer_access[][src_idx, dst_idx] = -1
743 end
744 end
745
746 return peer_access[][src_idx, dst_idx]
747 end
