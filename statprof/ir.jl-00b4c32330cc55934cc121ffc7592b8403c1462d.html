36 function basic_blocks_starts(stmts::Vector{Any})
37 jump_dests = BitSet()
38 push!(jump_dests, 1) # function entry point
39 # First go through and compute jump destinations
40 for idx in 1:length(stmts)
41 stmt = stmts[idx]
42 # Terminators
43 if isa(stmt, GotoIfNot)
44 push!(jump_dests, idx+1)
45 push!(jump_dests, stmt.dest)
46 elseif isa(stmt, ReturnNode)
47 idx < length(stmts) && push!(jump_dests, idx+1)
48 elseif isa(stmt, GotoNode)
49 # This is a fake dest to force the next stmt to start a bb
50 idx < length(stmts) && push!(jump_dests, idx+1)
51 push!(jump_dests, stmt.label)
52 elseif isa(stmt, Expr)
53 if stmt.head === :leave
54 # :leave terminates a BB
55 push!(jump_dests, idx+1)
56 elseif stmt.head === :enter
57 # :enter starts/ends a BB
58 push!(jump_dests, idx)
59 push!(jump_dests, idx+1)
60 # The catch block is a jump dest
61 push!(jump_dests, stmt.args[1]::Int)
62 end
63 end
64 if isa(stmt, PhiNode)
65 for edge in stmt.edges
66 if edge == idx - 1
67 push!(jump_dests, idx)
68 end
69 end
70 end
71 end
72 # and add add one more basic block start after the last statement
73 for i = length(stmts):-1:1
74 if stmts[i] !== nothing
75 push!(jump_dests, i+1)
76 break
77 end
78 end
79 return jump_dests
80 end

82 function compute_basic_blocks(stmts::Vector{Any})
83 bb_starts = basic_blocks_starts(stmts)
84 # Compute ranges
85 pop!(bb_starts, 1)
86 basic_block_index = collect(bb_starts)
87 blocks = BasicBlock[]
88 sizehint!(blocks, length(basic_block_index))
89 let first = 1
90 for last in basic_block_index
91 push!(blocks, BasicBlock(StmtRange(first, last - 1)))
92 first = last
93 end
94 end
95 # Compute successors/predecessors
96 for (num, b) in enumerate(blocks)
97 terminator = stmts[last(b.stmts)]
98 if isa(terminator, ReturnNode)
99 # return never has any successors
100 continue
101 end
102 if isa(terminator, GotoNode)
103 block′ = block_for_inst(basic_block_index, terminator.label)
104 push!(blocks[block′].preds, num)
105 push!(b.succs, block′)
106 continue
107 end
108 # Conditional Branch
109 if isa(terminator, GotoIfNot)
110 block′ = block_for_inst(basic_block_index, terminator.dest)
111 if block′ == num + 1
112 # This GotoIfNot acts like a noop - treat it as such.
113 # We will drop it during SSA renaming
114 else
115 push!(blocks[block′].preds, num)
116 push!(b.succs, block′)
117 end
118 elseif isa(terminator, Expr)
119 if terminator.head === :enter
120 # :enter gets a virtual edge to the exception handler and
121 # the exception handler gets a virtual edge from outside
122 # the function.
123 # See the devdocs on exception handling in SSA form (or
124 # bug Keno to write them, if you're reading this and they
125 # don't exist)
126 block′ = block_for_inst(basic_block_index, terminator.args[1]::Int)
127 push!(blocks[block′].preds, num)
128 push!(blocks[block′].preds, 0)
129 push!(b.succs, block′)
130 end
131 end
132 # statement fall-through
133 if num + 1 <= length(blocks)
134 push!(blocks[num + 1].preds, num)
135 push!(b.succs, num + 1)
136 end
137 end
138 return CFG(blocks, basic_block_index)
139 end

706 function insert_node!(compact::IncrementalCompact, before, inst::NewInstruction, attach_after::Bool=false)
707 @assert inst.effect_free_computed
708 if isa(before, SSAValue)
709 if before.id < compact.result_idx
710 count_added_node!(compact, inst.stmt)
711 line = something(inst.line, compact.result[before.id][:line])
712 node = add!(compact.new_new_nodes, before.id, attach_after)
713 node[:inst], node[:type], node[:line], node[:flag] = inst.stmt, inst.type, line, inst.flag
714 return NewSSAValue(node.idx)
715 else
716 line = something(inst.line, compact.ir.stmts[before.id][:line])
717 node = add_pending!(compact, before.id, attach_after)
718 node[:inst], node[:type], node[:line], node[:flag] = inst.stmt, inst.type, line, inst.flag
719 os = OldSSAValue(length(compact.ir.stmts) + length(compact.ir.new_nodes) + length(compact.pending_nodes))
720 push!(compact.ssa_rename, os)
721 push!(compact.used_ssas, 0)
722 return os
723 end
724 elseif isa(before, OldSSAValue)
725 pos = before.id
726 if pos > length(compact.ir.stmts)
727 #@assert attach_after
728 info = compact.pending_nodes.info[pos - length(compact.ir.stmts) - length(compact.ir.new_nodes)]
729 pos, attach_after = info.pos, info.attach_after
730 end
731 line = something(inst.line, compact.ir.stmts[pos][:line])
732 node = add_pending!(compact, pos, attach_after)
733 node[:inst], node[:type], node[:line], node[:flag] = inst.stmt, inst.type, line, inst.flag
734 os = OldSSAValue(length(compact.ir.stmts) + length(compact.ir.new_nodes) + length(compact.pending_nodes))
735 push!(compact.ssa_rename, os)
736 push!(compact.used_ssas, 0)
737 return os
738 elseif isa(before, NewSSAValue)
739 before_entry = compact.new_new_nodes.info[before.id]
740 line = something(inst.line, compact.new_new_nodes.stmts[before.id][:line])
741 new_entry = add!(compact.new_new_nodes, before_entry.pos, attach_after)
742 new_entry[:inst], new_entry[:type], new_entry[:line], new_entry[:flag] = inst.stmt, inst.type, line, inst.flag
743 return NewSSAValue(new_entry.idx)
744 else
745 error("Unsupported")
746 end
747 end

749 function insert_node_here!(compact::IncrementalCompact, inst::NewInstruction, reverse_affinity::Bool=false)
750 @assert inst.line !== nothing
751 refinish = false
752 result_idx = compact.result_idx
753 if reverse_affinity &&
754 ((compact.active_result_bb == length(compact.result_bbs) + 1) ||
755 result_idx == first(compact.result_bbs[compact.active_result_bb].stmts))
756 compact.active_result_bb -= 1
757 refinish = true
758 end
759 if result_idx > length(compact.result)
760 @assert result_idx == length(compact.result) + 1
761 resize!(compact, result_idx)
762 end
763 flag = inst.flag
764 if !inst.effect_free_computed && stmt_effect_free(inst.stmt, inst.type, compact, compact.ir.sptypes)
765 flag |= IR_FLAG_EFFECT_FREE
766 end
767 node = compact.result[result_idx]
768 node[:inst], node[:type], node[:line], node[:flag] = inst.stmt, inst.type, inst.line, flag
769 if count_added_node!(compact, inst.stmt)
770 push!(compact.late_fixup, result_idx)
771 end
772 compact.result_idx = result_idx + 1
773 inst = SSAValue(result_idx)
774 refinish && finish_current_bb!(compact, 0)
775 return inst
776 end

849 function process_phinode_values(old_values::Vector{Any}, late_fixup::Vector{Int},
850 processed_idx::Int, result_idx::Int,
851 ssa_rename::Vector{Any}, used_ssas::Vector{Int},
852 do_rename_ssa::Bool)
853 values = Vector{Any}(undef, length(old_values))
854 for i = 1:length(old_values)
855 isassigned(old_values, i) || continue
856 val = old_values[i]
857 if isa(val, SSAValue)
858 if do_rename_ssa
859 if val.id > processed_idx
860 push!(late_fixup, result_idx)
861 val = OldSSAValue(val.id)
862 else
863 val = renumber_ssa2(val, ssa_rename, used_ssas, do_rename_ssa)
864 end
865 else
866 used_ssas[val.id] += 1
867 end
868 elseif isa(val, OldSSAValue)
869 if val.id > processed_idx
870 push!(late_fixup, result_idx)
871 else
872 # Always renumber these. do_rename_ssa applies only to actual SSAValues
873 val = renumber_ssa2(SSAValue(val.id), ssa_rename, used_ssas, true)
874 end
875 elseif isa(val, NewSSAValue)
876 push!(late_fixup, result_idx)
877 end
878 values[i] = val
879 end
880 return values
881 end

883 function renumber_ssa2(val::SSAValue, ssanums::Vector{Any}, used_ssas::Vector{Int}, do_rename_ssa::Bool)
884 id = val.id
885 if id > length(ssanums)
886 return val
887 end
888 if do_rename_ssa
889 val = ssanums[id]
890 end
891 if isa(val, SSAValue)
892 if used_ssas !== nothing
893 used_ssas[val.id] += 1
894 end
895 end
896 return val

899 function renumber_ssa2!(@nospecialize(stmt), ssanums::Vector{Any}, used_ssas::Vector{Int}, late_fixup::Vector{Int}, result_idx::Int, do_rename_ssa::Bool)
900 urs = userefs(stmt)
901 for op in urs
902 val = op[]
903 if isa(val, OldSSAValue) || isa(val, NewSSAValue)
904 push!(late_fixup, result_idx)
905 end
906 if isa(val, SSAValue)
907 val = renumber_ssa2(val, ssanums, used_ssas, do_rename_ssa)
908 end
909 if isa(val, OldSSAValue) || isa(val, NewSSAValue)
910 push!(late_fixup, result_idx)
911 end
912 op[] = val
913 end
914 return urs[]

917 # Used in inlining before we start compacting - Only works at the CFG level
918 function kill_edge!(bbs::Vector{BasicBlock}, from::Int, to::Int)
919 preds, succs = bbs[to].preds, bbs[from].succs
920 deleteat!(preds, findfirst(x->x === from, preds)::Int)
921 deleteat!(succs, findfirst(x->x === to, succs)::Int)
922 if length(preds) == 0
923 for succ in copy(bbs[to].succs)
924 kill_edge!(bbs, to, succ)
925 end
926 end
927 end

929 # N.B.: from and to are non-renamed indices
930 function kill_edge!(compact::IncrementalCompact, active_bb::Int, from::Int, to::Int)
931 # Note: We recursively kill as many edges as are obviously dead. However, this
932 # may leave dead loops in the IR. We kill these later in a CFG cleanup pass (or
933 # worstcase during codegen).
934 preds = compact.result_bbs[compact.bb_rename_succ[to]].preds
935 succs = compact.result_bbs[compact.bb_rename_pred[from]].succs
936 deleteat!(preds, findfirst(x->x === compact.bb_rename_pred[from], preds)::Int)
937 deleteat!(succs, findfirst(x->x === compact.bb_rename_succ[to], succs)::Int)
938 # Check if the block is now dead
939 if length(preds) == 0
940 for succ in copy(compact.result_bbs[compact.bb_rename_succ[to]].succs)
941 kill_edge!(compact, active_bb, to, findfirst(x->x === succ, compact.bb_rename_pred)::Int)
942 end
943 if to < active_bb
944 # Kill all statements in the block
945 stmts = compact.result_bbs[compact.bb_rename_succ[to]].stmts
946 for stmt in stmts
947 compact.result[stmt][:inst] = nothing
948 end
949 compact.result[last(stmts)][:inst] = ReturnNode()
950 end
951 else
952 # Remove this edge from all phi nodes in `to` block
953 # NOTE: It is possible for `to` to contain only `nothing` statements,
954 # so we must be careful to stop at its last statement
955 if to < active_bb
956 stmts = compact.result_bbs[compact.bb_rename_succ[to]].stmts
957 idx = first(stmts)
958 while idx <= last(stmts)
959 stmt = compact.result[idx][:inst]
960 stmt === nothing && continue
961 isa(stmt, PhiNode) || break
962 i = findfirst(x-> x == compact.bb_rename_pred[from], stmt.edges)
963 if i !== nothing
964 deleteat!(stmt.edges, i)
965 deleteat!(stmt.values, i)
966 end
967 idx += 1
968 end
969 else
970 stmts = compact.ir.cfg.blocks[to].stmts
971 for stmt in CompactPeekIterator(compact, first(stmts), last(stmts))
972 stmt === nothing && continue
973 isa(stmt, PhiNode) || break
974 i = findfirst(x-> x == from, stmt.edges)
975 if i !== nothing
976 deleteat!(stmt.edges, i)
977 deleteat!(stmt.values, i)
978 end
979 end
980 end
981 end
982 nothing
983 end

985 function process_node!(compact::IncrementalCompact, result_idx::Int, inst::Instruction, idx::Int, processed_idx::Int, active_bb::Int, do_rename_ssa::Bool)
986 stmt = inst[:inst]
987 result = compact.result
988 ssa_rename = compact.ssa_rename
989 late_fixup = compact.late_fixup
990 used_ssas = compact.used_ssas
991 ssa_rename[idx] = SSAValue(result_idx)
992 if stmt === nothing
993 ssa_rename[idx] = stmt
994 elseif isa(stmt, OldSSAValue)
995 ssa_rename[idx] = ssa_rename[stmt.id]
996 elseif isa(stmt, GotoNode) && compact.cfg_transforms_enabled
997 result[result_idx][:inst] = GotoNode(compact.bb_rename_succ[stmt.label])
998 result_idx += 1
999 elseif isa(stmt, GlobalRef) || isa(stmt, GotoNode)
1000 result[result_idx][:inst] = stmt
1001 result_idx += 1
1002 elseif isa(stmt, GotoIfNot) && compact.cfg_transforms_enabled
1003 stmt = renumber_ssa2!(stmt, ssa_rename, used_ssas, late_fixup, result_idx, do_rename_ssa)::GotoIfNot
1004 result[result_idx][:inst] = stmt
1005 cond = stmt.cond
1006 if isa(cond, Bool) && compact.fold_constant_branches
1007 if cond
1008 result[result_idx][:inst] = nothing
1009 kill_edge!(compact, active_bb, active_bb, stmt.dest)
1010 # Don't increment result_idx => Drop this statement
1011 else
1012 result[result_idx][:inst] = GotoNode(compact.bb_rename_succ[stmt.dest])
1013 kill_edge!(compact, active_bb, active_bb, active_bb+1)
1014 result_idx += 1
1015 end
1016 else
1017 result[result_idx][:inst] = GotoIfNot(cond, compact.bb_rename_succ[stmt.dest])
1018 result_idx += 1
1019 end
1020 elseif isa(stmt, Expr)
1021 stmt = renumber_ssa2!(stmt, ssa_rename, used_ssas, late_fixup, result_idx, do_rename_ssa)::Expr
1022 if compact.cfg_transforms_enabled && isexpr(stmt, :enter)
1023 stmt.args[1] = compact.bb_rename_succ[stmt.args[1]::Int]
1024 end
1025 result[result_idx][:inst] = stmt
1026 result_idx += 1
1027 elseif isa(stmt, PiNode)
1028 # As an optimization, we eliminate any trivial pinodes. For performance, we use ===
1029 # type equality. We may want to consider using == in either a separate pass or if
1030 # performance turns out ok
1031 stmt = renumber_ssa2!(stmt, ssa_rename, used_ssas, late_fixup, result_idx, do_rename_ssa)::PiNode
1032 pi_val = stmt.val
1033 if isa(pi_val, SSAValue)
1034 if stmt.typ === compact.result[pi_val.id][:type]
1035 ssa_rename[idx] = pi_val
1036 return result_idx
1037 end
1038 elseif isa(pi_val, Argument)
1039 if stmt.typ === compact.ir.argtypes[pi_val.n]
1040 ssa_rename[idx] = pi_val
1041 return result_idx
1042 end
1043 elseif !isa(pi_val, AnySSAValue) && !isa(pi_val, GlobalRef)
1044 valtyp = isa(pi_val, QuoteNode) ? typeof(pi_val.value) : typeof(pi_val)
1045 if valtyp === stmt.typ
1046 ssa_rename[idx] = pi_val
1047 return result_idx
1048 end
1049 end
1050 result[result_idx][:inst] = stmt
1051 result_idx += 1
1052 elseif isa(stmt, ReturnNode) || isa(stmt, UpsilonNode) || isa(stmt, GotoIfNot)
1053 result[result_idx][:inst] = renumber_ssa2!(stmt, ssa_rename, used_ssas, late_fixup, result_idx, do_rename_ssa)
1054 result_idx += 1
1055 elseif isa(stmt, PhiNode)
1056 if compact.cfg_transforms_enabled
1057 # Rename phi node edges
1058 map!(i -> compact.bb_rename_pred[i], stmt.edges, stmt.edges)
1059
1060 # Remove edges and values associated with dead blocks. Entries in
1061 # `values` can be undefined when the phi node refers to something
1062 # that is not defined. (This is not a sign that the compiler is
1063 # unintentionally leaving some entries in `values` uninitialized.)
1064 # For example, consider a reference to a variable that is only
1065 # defined if some branch is taken.
1066 #
1067 # In order to leave undefined values undefined (undefined-ness is
1068 # not a value we can copy), we copy only the edges and (defined)
1069 # values we want to keep to new arrays initialized with undefined
1070 # elements.
1071 edges = Vector{Int32}(undef, length(stmt.edges))
1072 values = Vector{Any}(undef, length(stmt.values))
1073 new_index = 1
1074 for old_index in 1:length(stmt.edges)
1075 if stmt.edges[old_index] != -1
1076 edges[new_index] = stmt.edges[old_index]
1077 if isassigned(stmt.values, old_index)
1078 values[new_index] = stmt.values[old_index]
1079 end
1080 new_index += 1
1081 end
1082 end
1083 resize!(edges, new_index-1)
1084 resize!(values, new_index-1)
1085 else
1086 edges = stmt.edges
1087 values = stmt.values
1088 end
1089
1090 values = process_phinode_values(values, late_fixup, processed_idx, result_idx, ssa_rename, used_ssas, do_rename_ssa)
1091 # Don't remove the phi node if it is before the definition of its value
1092 # because doing so can create forward references. This should only
1093 # happen with dead loops, but can cause problems when optimization
1094 # passes look at all code, dead or not. This check should be
1095 # unnecessary when DCE can remove those dead loops entirely, so this is
1096 # just to be safe.
1097 before_def = isassigned(values, 1) && isa(values[1], OldSSAValue) &&
1098 idx < values[1].id
1099 if length(edges) == 1 && isassigned(values, 1) && !before_def &&
1100 length(compact.cfg_transforms_enabled ?
1101 compact.result_bbs[compact.bb_rename_succ[active_bb]].preds :
1102 compact.ir.cfg.blocks[active_bb].preds) == 1
1103 # There's only one predecessor left - just replace it
1104 ssa_rename[idx] = values[1]
1105 else
1106 result[result_idx][:inst] = PhiNode(edges, values)
1107 result_idx += 1
1108 end
1109 elseif isa(stmt, PhiCNode)
1110 result[result_idx][:inst] = PhiCNode(process_phinode_values(stmt.values, late_fixup, processed_idx, result_idx, ssa_rename, used_ssas, do_rename_ssa))
1111 result_idx += 1
1112 elseif isa(stmt, SSAValue)
1113 # identity assign, replace uses of this ssa value with its result
1114 if do_rename_ssa
1115 stmt = ssa_rename[stmt.id]
1116 end
1117 ssa_rename[idx] = stmt
1118 else
1119 # Constant assign, replace uses of this ssa value with its result
1120 ssa_rename[idx] = stmt
1121 end
1122 return result_idx
1123 end

1135 function finish_current_bb!(compact::IncrementalCompact, active_bb, old_result_idx=compact.result_idx, unreachable=false)
1136 if compact.active_result_bb > length(compact.result_bbs)
1137 #@assert compact.bb_rename[active_bb] == -1
1138 return true
1139 end
1140 bb = compact.result_bbs[compact.active_result_bb]
1141 # If this was the last statement in the BB and we decided to skip it, insert a
1142 # dummy `nothing` node, to prevent changing the structure of the CFG
1143 skipped = false
1144 if !compact.cfg_transforms_enabled || active_bb == 0 || active_bb > length(compact.bb_rename_succ) || compact.bb_rename_succ[active_bb] != -1
1145 if compact.result_idx == first(bb.stmts)
1146 length(compact.result) < old_result_idx && resize!(compact, old_result_idx)
1147 node = compact.result[old_result_idx]
1148 if unreachable
1149 node[:inst], node[:type], node[:line] = ReturnNode(), Union{}, 0
1150 else
1151 node[:inst], node[:type], node[:line] = nothing, Nothing, 0
1152 end
1153 compact.result_idx = old_result_idx + 1
1154 elseif compact.cfg_transforms_enabled && compact.result_idx - 1 == first(bb.stmts)
1155 # Optimization: If this BB consists of only a branch, eliminate this bb
1156 end
1157 compact.result_bbs[compact.active_result_bb] = BasicBlock(bb, StmtRange(first(bb.stmts), compact.result_idx-1))
1158 compact.active_result_bb += 1
1159 else
1160 skipped = true
1161 end
1162 if compact.active_result_bb <= length(compact.result_bbs)
1163 new_bb = compact.result_bbs[compact.active_result_bb]
1164 compact.result_bbs[compact.active_result_bb] = BasicBlock(new_bb,
1165 StmtRange(compact.result_idx, last(new_bb.stmts)))
1166 end
1167 return skipped
1168 end

1171 compact.new_nodes_idx > length(compact.perm) && return false
1172 entry = compact.ir.new_nodes.info[compact.perm[compact.new_nodes_idx]]

1176 function process_newnode!(compact::IncrementalCompact, new_idx::Int, new_node_entry::Instruction, new_node_info::NewNodeInfo, idx::Int, active_bb::Int, do_rename_ssa::Bool)
1177 old_result_idx = compact.result_idx
1178 bb = compact.ir.cfg.blocks[active_bb]
1179 node = compact.result[old_result_idx]
1180 node[] = new_node_entry
1181 result_idx = process_node!(compact, old_result_idx, node, new_idx, idx - 1, active_bb, do_rename_ssa)
1182 compact.result_idx = result_idx
1183 # If this instruction has reverse affinity and we were at the end of a basic block,
1184 # finish it now.
1185 if new_node_info.attach_after && idx == last(bb.stmts)+1 && !attach_after_stmt_after(compact, idx-1)
1186 active_bb += 1
1187 finish_current_bb!(compact, active_bb, old_result_idx)
1188 end
1189 (old_result_idx == result_idx) && return iterate(compact, (idx, active_bb))
1190 return Pair{Pair{Int, Int}, Any}(
1191 Pair{Int,Int}(new_idx,old_result_idx),
1192 compact.result[old_result_idx][:inst]), (idx, active_bb)
1193 end
