61 _time_ns() = ccall(:jl_hrtime, UInt64, ()) # Re-implemented here because Base not yet available.

63 # We keep a stack of the Timings for each of the MethodInstances currently being timed.
64 # Since type inference currently operates via a depth-first search (during abstract
65 # evaluation), this vector operates like a call stack. The last node in _timings is the
66 # node currently being inferred, and its parent is directly before it, etc.
67 # Each Timing also contains its own vector for all of its children, so that the tree
68 # call structure through type inference is recorded. (It's recorded as a tree, not a graph,
69 # because we create a new node for duplicates.)
70 const _timings = Timing[]
71 # ROOT() is an empty function used as the top-level Timing node to measure all time spent
72 # *not* in type inference during a given recording trace. It is used as a "dummy" node.
73 function ROOT() end

97 @inline function close_current_timer()

100 accum_time = stop_time - parent_timer.cur_start_time

102 # Add in accum_time ("modify" the immutable struct)

116 @inline function enter_new_timer(frame)
117 # Very first thing, stop the active timer: get the current time and add in the
118 # time since it was last started to its aggregate exclusive time.
119 close_current_timer()
120
121 mi_info = _typeinf_identifier(frame)
122
123 # Start the new timer right before returning
124 push!(_timings, Timing(mi_info, UInt64(0)))
125 len = length(_timings)
126 new_timer = @inbounds _timings[len]
127 # Set the current time _after_ appending the node, to try to exclude the
128 # overhead from measurement.
129 start = _time_ns()
130
131 @inbounds begin
132 _timings[len] = Timing(
133 new_timer.mi_info,
134 start,
135 start,
136 new_timer.time,
137 new_timer.children,
138 )
139 end

142 end

144 # _expected_frame_ is not needed within this function; it is used in the `@assert`, to
145 # assert that indeed we are always returning to a parent after finishing all of its
146 # children (that is, asserting that inference proceeds via depth-first-search).
147 @inline function exit_current_timer(_expected_frame_)
148 # Finish the new timer
149 stop_time = _time_ns()
150
151 expected_mi_info = _typeinf_identifier(_expected_frame_)
152
153 # Grab the new timer again because it might have been modified in _timings
154 # (since it's an immutable struct)
155 # And remove it from the current timings stack
156 new_timer = pop!(_timings)
157 Core.Compiler.@assert new_timer.mi_info.mi === expected_mi_info.mi
158
159 # Prepare to unwind one level of the stack and record in the parent
160 parent_timer = _timings[end]
161
162 accum_time = stop_time - new_timer.cur_start_time
163 # Add in accum_time ("modify" the immutable struct)
164 new_timer = Timing(
165 new_timer.mi_info,
166 new_timer.start_time,
167 new_timer.cur_start_time,
168 new_timer.time + accum_time,
169 new_timer.children,
170 parent_timer.mi_info.mi === ROOTmi ? backtrace() : nothing,
171 )
172 # Record the final timing with the original parent timer
173 push!(parent_timer.children, new_timer)
174
175 # And finally restart the parent timer:
176 len = length(_timings)
177 @inbounds begin
178 _timings[len] = Timing(
179 parent_timer.mi_info,
180 parent_timer.start_time,
181 _time_ns(),
182 parent_timer.time,
183 parent_timer.children,
184 parent_timer.bt,
185 )
186 end
