56 function issorted(itr, order::Ordering)
57 y = iterate(itr)
58 y === nothing && return true

60 y = iterate(itr, state)
61 while y !== nothing
62 this, state = y
63 lt(order, this, prev) && return false
64 prev = this
65 y = iterate(itr, state)
66 end
67 return true
68 end

519 # selectpivot!
520 #
521 # Given 3 locations in an array (lo, mi, and hi), sort v[lo], v[mi], v[hi]) and
522 # choose the middle value as a pivot
523 #
524 # Upon return, the pivot is in v[lo], and v[hi] is guaranteed to be
525 # greater than the pivot
526
527 @inline function selectpivot!(v::AbstractVector, lo::Integer, hi::Integer, o::Ordering)
528 @inbounds begin
529 mi = midpoint(lo, hi)
530
531 # sort v[mi] <= v[lo] <= v[hi] such that the pivot is immediately in place
532 if lt(o, v[lo], v[mi])
533 v[mi], v[lo] = v[lo], v[mi]
534 end
535
536 if lt(o, v[hi], v[lo])
537 if lt(o, v[hi], v[mi])
538 v[hi], v[lo], v[mi] = v[lo], v[mi], v[hi]
539 else
540 v[hi], v[lo] = v[lo], v[hi]
541 end
542 end
543
544 # return the pivot
545 return v[lo]
546 end
547 end

549 # partition!
550 #
551 # select a pivot, and partition v according to the pivot
552
553 function partition!(v::AbstractVector, lo::Integer, hi::Integer, o::Ordering)
554 pivot = selectpivot!(v, lo, hi, o)
555 # pivot == v[lo], v[hi] > pivot
556 i, j = lo, hi
557 @inbounds while true
558 i += 1; j -= 1
559 while lt(o, v[i], pivot); i += 1; end;
560 while lt(o, pivot, v[j]); j -= 1; end;
561 i >= j && break
562 v[i], v[j] = v[j], v[i]
563 end
564 v[j], v[lo] = pivot, v[j]
565
566 # v[j] == pivot
567 # v[k] >= pivot for k > j
568 # v[i] <= pivot for i < j
569 return j
570 end

665 """
666 sort!(v; alg::Algorithm=defalg(v), lt=isless, by=identity, rev::Bool=false, order::Ordering=Forward)
667
668 Sort the vector `v` in place. [`QuickSort`](@ref) is used by default for numeric arrays while
669 [`MergeSort`](@ref) is used for other arrays. You can specify an algorithm to use via the `alg`
670 keyword (see [Sorting Algorithms](@ref) for available algorithms). The `by` keyword lets you provide
671 a function that will be applied to each element before comparison; the `lt` keyword allows
672 providing a custom "less than" function (note that for every `x` and `y`, only one of `lt(x,y)`
673 and `lt(y,x)` can return `true`); use `rev=true` to reverse the sorting order. These
674 options are independent and can be used together in all possible combinations: if both `by`
675 and `lt` are specified, the `lt` function is applied to the result of the `by` function;
676 `rev=true` reverses whatever ordering specified via the `by` and `lt` keywords.
677

681 3-element Vector{Int64}:

683 2
684 3

687 3-element Vector{Int64}:
688 3
689 2
690 1
