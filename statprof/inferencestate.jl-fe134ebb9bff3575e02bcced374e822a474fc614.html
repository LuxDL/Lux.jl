122 function compute_trycatch(code::Vector{Any}, ip::BitSet)
123 # The goal initially is to record the frame like this for the state at exit:
124 # 1: (enter 3) # == 0
125 # 3: (expr) # == 1
126 # 3: (leave 1) # == 1
127 # 4: (expr) # == 0
128 # then we can find all trys by walking backwards from :enter statements,
129 # and all catches by looking at the statement after the :enter
130 n = length(code)
131 empty!(ip)
132 ip.offset = 0 # for _bits_findnext
133 push!(ip, n + 1)
134 handler_at = fill(0, n)
135
136 # start from all :enter statements and record the location of the try
137 for pc = 1:n
138 stmt = code[pc]
139 if isexpr(stmt, :enter)
140 l = stmt.args[1]::Int
141 handler_at[pc + 1] = pc
142 push!(ip, pc + 1)
143 handler_at[l] = pc
144 push!(ip, l)
145 end
146 end
147
148 # now forward those marks to all :leave statements
149 pc´´ = 0
150 while true
151 # make progress on the active ip set
152 pc = _bits_findnext(ip.bits, pc´´)::Int
153 pc > n && break
154 while true # inner loop optimizes the common case where it can run straight from pc to pc + 1
155 pc´ = pc + 1 # next program-counter (after executing instruction)
156 if pc == pc´´
157 pc´´ = pc´
158 end
159 delete!(ip, pc)
160 cur_hand = handler_at[pc]
161 @assert cur_hand != 0 "unbalanced try/catch"
162 stmt = code[pc]
163 if isa(stmt, GotoNode)
164 pc´ = stmt.label
165 elseif isa(stmt, GotoIfNot)
166 l = stmt.dest::Int
167 if handler_at[l] != cur_hand
168 @assert handler_at[l] == 0 "unbalanced try/catch"
169 handler_at[l] = cur_hand
170 if l < pc´´
171 pc´´ = l
172 end
173 push!(ip, l)
174 end
175 elseif isa(stmt, ReturnNode)
176 @assert !isdefined(stmt, :val) "unbalanced try/catch"
177 break
178 elseif isa(stmt, Expr)
179 head = stmt.head
180 if head === :enter
181 cur_hand = pc
182 elseif head === :leave
183 l = stmt.args[1]::Int
184 for i = 1:l
185 cur_hand = handler_at[cur_hand]
186 end
187 cur_hand == 0 && break
188 end
189 end
190
191 pc´ > n && break # can't proceed with the fast-path fall-through
192 if handler_at[pc´] != cur_hand
193 @assert handler_at[pc´] == 0 "unbalanced try/catch"
194 handler_at[pc´] = cur_hand
195 elseif !in(pc´, ip)
196 break # already visited
197 end
198 pc = pc´
199 end
200 end
201
202 @assert first(ip) == n + 1
203 return handler_at
204 end

208 Iterate through all callers of the given InferenceState in the abstract
209 interpretation stack (including the given InferenceState itself), vising
210 children before their parents (i.e. ascending the tree from the given
211 InferenceState). Note that cycles may be visited in any order.
212 """

215 end
216 iterate(unw::InfStackUnwind) = (unw.inf, (unw.inf, 0))
217 function iterate(unw::InfStackUnwind, (infstate, cyclei)::Tuple{InferenceState, Int})
218 # iterate through the cycle before walking to the parent
219 if cyclei < length(infstate.callers_in_cycle)
220 cyclei += 1
221 infstate = infstate.callers_in_cycle[cyclei]
222 else
223 cyclei = 0
224 infstate = infstate.parent
225 end
226 infstate === nothing && return nothing
227 (infstate::InferenceState, (infstate, cyclei))
228 end

240 function sptypes_from_meth_instance(linfo::MethodInstance)
241 toplevel = !isa(linfo.def, Method)
242 if !toplevel && isempty(linfo.sparam_vals) && isa(linfo.def.sig, UnionAll)
243 # linfo is unspecialized
244 sp = Any[]
245 sig = linfo.def.sig
246 while isa(sig, UnionAll)
247 push!(sp, sig.var)
248 sig = sig.body
249 end
250 else
251 sp = collect(Any, linfo.sparam_vals)
252 end
253 for i = 1:length(sp)
254 v = sp[i]
255 if v isa TypeVar
256 fromArg = 0
257 # if this parameter came from arg::Type{T}, then `arg` is more precise than
258 # Type{T} where lb<:T<:ub
259 sig = linfo.def.sig
260 temp = sig
261 for j = 1:i-1
262 temp = temp.body
263 end
264 Pi = temp.var
265 while temp isa UnionAll
266 temp = temp.body
267 end
268 sigtypes = (temp::DataType).parameters
269 for j = 1:length(sigtypes)
270 tj = sigtypes[j]
271 if isType(tj) && tj.parameters[1] === Pi
272 fromArg = j
273 break
274 end
275 end
276 if fromArg > 0
277 ty = fieldtype(linfo.specTypes, fromArg)
278 else
279 ub = v.ub
280 while ub isa TypeVar
281 ub = ub.ub
282 end
283 if has_free_typevars(ub)
284 ub = Any
285 end
286 lb = v.lb
287 while lb isa TypeVar
288 lb = lb.lb
289 end
290 if has_free_typevars(lb)
291 lb = Bottom
292 end
293 if Any <: ub && lb <: Bottom
294 ty = Any
295 else
296 tv = TypeVar(v.name, lb, ub)
297 ty = UnionAll(tv, Type{tv})
298 end
299 end
300 elseif isvarargtype(v)
301 ty = Int
302 else
303 ty = Const(v)
304 end
305 sp[i] = ty
306 end
307 return sp
308 end

321 function record_ssa_assign(ssa_id::Int, @nospecialize(new), frame::InferenceState)
322 old = frame.src.ssavaluetypes[ssa_id]
323 if old === NOT_FOUND || !(new ⊑ old)
324 # typically, we expect that old ⊑ new (that output information only
325 # gets less precise with worse input information), but to actually
326 # guarantee convergence we need to use tmerge here to ensure that is true
327 frame.src.ssavaluetypes[ssa_id] = old === NOT_FOUND ? new : tmerge(old, new)
328 W = frame.ip
329 s = frame.stmt_types
330 for r in frame.ssavalue_uses[ssa_id]
331 if s[r] !== nothing # s[r] === nothing => unreached statement
332 if r < frame.pc´´
333 frame.pc´´ = r
334 end
335 push!(W, r)
336 end
337 end
338 end
339 nothing
340 end
