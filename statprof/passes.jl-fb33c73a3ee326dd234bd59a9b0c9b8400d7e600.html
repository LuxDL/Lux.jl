3 """
4 This struct keeps track of all uses of some mutable struct allocated
5 in the current function. `uses` are all instances of `getfield` on the
6 struct. `defs` are all instances of `setfield!` on the struct. The terminology
7 refers to the uses/defs of the ``slot bundle'' that the mutable struct represents.
8
9 In addition we keep track of all instances of a foreigncall preserve of this mutable
10 struct. Somewhat counterintuitively, we don't actually need to make sure that the
11 struct itself is live (or even allocated) at a ccall site. If there are no other places
12 where the struct escapes (and thus e.g. where its address is taken), it need not be
13 allocated. We do however, need to make sure to preserve any elements of this struct.
14 """
15 struct SSADefUse
16 uses::Vector{Int}
17 defs::Vector{Int}
18 ccall_preserve_uses::Vector{Int}
19 end

58 function compute_value_for_use(ir::IRCode, domtree::DomTree, allblocks::Vector{Int}, du::SSADefUse, phinodes::IdDict{Int, SSAValue}, fidx::Int, use_idx::Int)
59 # Find the first dominating def
60 curblock = stmtblock = block_for_inst(ir.cfg, use_idx)
61 curblock = find_curblock(domtree, allblocks, curblock)
62 defblockdefs = let curblock = curblock
63 Int[stmt for stmt in du.defs if block_for_inst(ir.cfg, stmt) == curblock]
64 end
65 def = 0
66 if !isempty(defblockdefs)
67 if curblock != stmtblock
68 # Find the last def in this block
69 def = 0
70 for x in defblockdefs
71 def = max(def, x)
72 end
73 else
74 # Find the last def before our use
75 def = 0
76 for x in defblockdefs
77 def = max(def, x >= use_idx ? 0 : x)
78 end
79 end
80 end
81 if def == 0
82 if !haskey(phinodes, curblock)
83 # If this happens, we need to search the predecessors for defs. Which
84 # one doesn't matter - if it did, we'd have had a phinode
85 return compute_value_for_block(ir, domtree, allblocks, du, phinodes, fidx, first(ir.cfg.blocks[stmtblock].preds))
86 end
87 # The use is the phinode
88 return phinodes[curblock]
89 else
90 return val_for_def_expr(ir, def, fidx)
91 end
92 end

449 function perform_lifting!(compact::IncrementalCompact,
450 visited_phinodes::Vector{Any}, @nospecialize(cache_key),
451 lifting_cache::IdDict{Pair{AnySSAValue, Any}, AnySSAValue},
452 @nospecialize(result_t), lifted_leaves::IdDict{Any, Any}, @nospecialize(stmt_val))
453 reverse_mapping = IdDict{Any, Any}(ssa => id for (id, ssa) in enumerate(visited_phinodes))
454
455 # Insert PhiNodes
456 lifted_phis = LiftedPhi[]
457 for item in visited_phinodes
458 if (item, cache_key) in keys(lifting_cache)
459 ssa = lifting_cache[Pair{AnySSAValue, Any}(item, cache_key)]
460 push!(lifted_phis, LiftedPhi(ssa, compact[ssa], false))
461 continue
462 end
463 n = PhiNode()
464 ssa = insert_node!(compact, item, effect_free(NewInstruction(n, result_t)))
465 lifting_cache[Pair{AnySSAValue, Any}(item, cache_key)] = ssa
466 push!(lifted_phis, LiftedPhi(ssa, n, true))
467 end
468
469 # Fix up arguments
470 for (old_node_ssa, lf) in zip(visited_phinodes, lifted_phis)
471 old_node = compact[old_node_ssa]
472 new_node = lf.node
473 lf.need_argupdate || continue
474 for i = 1:length(old_node.edges)
475 edge = old_node.edges[i]
476 isassigned(old_node.values, i) || continue
477 val = old_node.values[i]
478 orig_val = val
479 if is_old(compact, old_node_ssa) && isa(val, SSAValue)
480 val = OldSSAValue(val.id)
481 end
482 if isa(val, Union{NewSSAValue, SSAValue, OldSSAValue})
483 val = simple_walk(compact, val)
484 end
485 if val in keys(lifted_leaves)
486 push!(new_node.edges, edge)
487 lifted_val = lifted_leaves[val]
488 if lifted_val === nothing
489 resize!(new_node.values, length(new_node.values)+1)
490 continue
491 end
492 lifted_val = lifted_val.x
493 if isa(lifted_val, Union{NewSSAValue, SSAValue, OldSSAValue})
494 lifted_val = simple_walk(compact, lifted_val, (pi, idx)->true)
495 end
496 push!(new_node.values, lifted_val)
497 elseif isa(val, Union{NewSSAValue, SSAValue, OldSSAValue}) && val in keys(reverse_mapping)
498 push!(new_node.edges, edge)
499 push!(new_node.values, lifted_phis[reverse_mapping[val]].ssa)
500 else
501 # Probably ignored by path condition, skip this
502 end
503 end
504 end
505
506 for lf in lifted_phis
507 count_added_node!(compact, lf.node)
508 end
509
510 # Fixup the stmt itself
511 if isa(stmt_val, Union{SSAValue, OldSSAValue})
512 stmt_val = simple_walk(compact, stmt_val)
513 end
514
515 if stmt_val in keys(lifted_leaves)
516 stmt_val = lifted_leaves[stmt_val]
517 elseif isa(stmt_val, Union{NewSSAValue, SSAValue, OldSSAValue}) && stmt_val in keys(reverse_mapping)
518 stmt_val = RefValue{Any}(lifted_phis[reverse_mapping[stmt_val]].ssa)
519 end
520
521 return stmt_val
522 end

638 field = stmt.args[3]
639 isa(field, QuoteNode) && (field = field.value)
640 isa(field, Union{Int, Symbol}) || continue
641
642 struct_typ = unwrap_unionall(widenconst(compact_exprtype(compact, stmt.args[2])))
643 if isa(struct_typ, Union) && struct_typ <: Tuple
644 struct_typ = unswitchtupleunion(struct_typ)
645 end
646 isa(struct_typ, DataType) || continue
647
648 struct_typ.name.atomicfields == C_NULL || continue # TODO: handle more
649 if !(field_ordering === :unspecified || (field_ordering isa Const && field_ordering.val === :not_atomic))
650 continue
651 end
652
653 def, typeconstraint = stmt.args[2], struct_typ
654
655 if ismutabletype(struct_typ)
656 isa(def, SSAValue) || continue
657 let intermediaries = IdSet()
658 callback = function(@nospecialize(pi), ssa::AnySSAValue)
659 push!(intermediaries, ssa.id)
660 return false
661 end
662 def = simple_walk(compact, def, callback)
663 # Mutable stuff here
664 isa(def, SSAValue) || continue
665 mid, defuse = get!(defuses, def.id, (IdSet{Int}(), SSADefUse()))
666 if is_setfield
667 push!(defuse.defs, idx)
668 else
669 push!(defuse.uses, idx)
670 end
671 union!(mid, intermediaries)
672 end
673 continue
674 elseif is_setfield
675 continue
676 end
677
678 if isa(def, Union{OldSSAValue, SSAValue})
679 def, typeconstraint = simple_walk_constraint(compact, def, typeconstraint)
680 end
681
682 visited_phinodes = Any[]
683 leaves = walk_to_defs(compact, def, typeconstraint, visited_phinodes)
684
685 isempty(leaves) && continue
686
687 field = try_compute_fieldidx(struct_typ, field)
688 field === nothing && continue
689
690 r = lift_leaves(compact, stmt, result_t, field, leaves)
691 r === nothing && continue
692 lifted_leaves, any_undef = r
693
694 if any_undef
695 result_t = make_MaybeUndef(result_t)
696 end
697
698 # @Base.show result_t
699 # @Base.show stmt
700 # for (k,v) in lifted_leaves
701 # @Base.show (k, v)
702 # if isa(k, AnySSAValue)
703 # @Base.show compact[k]
704 # end
705 # if isa(v, RefValue) && isa(v.x, AnySSAValue)
706 # @Base.show compact[v.x]
707 # end
708 # end
709 val = perform_lifting!(compact, visited_phinodes, field, lifting_cache, result_t, lifted_leaves, stmt.args[2])
710
711 # Insert the undef check if necessary
712 if any_undef
713 if val === nothing
714 insert_node!(compact, SSAValue(idx),
715 non_effect_free(NewInstruction(Expr(:throw_undef_if_not, Symbol("##getfield##"), false), Nothing)))
716 else
717 # val must be defined
718 end
719 else
720 @assert val !== nothing
721 end
722
723 global assertion_counter
724 assertion_counter::Int += 1
725 #insert_node_here!(compact, Expr(:assert_egal, Symbol(string("assert_egal_", assertion_counter)), SSAValue(idx), val), nothing, 0, true)
726 #continue
727 compact[idx] = val === nothing ? nothing : val.x
728 end
729
730
731 non_dce_finish!(compact)
732 # Copy the use count, `simple_dce!` may modify it and for our predicate
733 # below we need it consistent with the state of the IR here (after tracking
734 # phi node arguments, but before dce).
735 used_ssas = copy(compact.used_ssas)
736 simple_dce!(compact)
737 ir = complete(compact)
738
739 # Compute domtree, needed below, now that we have finished compacting the
740 # IR. This needs to be after we iterate through the IR with
741 # `IncrementalCompact` because removing dead blocks can invalidate the
742 # domtree.
743 1 (0 %)
1 (100 %) samples spent calling construct_domtree
@timeit "domtree 2" domtree = construct_domtree(ir.cfg.blocks)
744
745 # Now go through any mutable structs and see which ones we can eliminate
746 for (idx, (intermediaries, defuse)) in defuses
747 intermediaries = collect(intermediaries)
748 # Check if there are any uses we did not account for. If so, the variable
749 # escapes and we cannot eliminate the allocation. This works, because we're guaranteed
750 # not to include any intermediaries that have dead uses. As a result, missing uses will only ever
751 # show up in the nuses_total count.
752 nleaves = length(defuse.uses) + length(defuse.defs) + length(defuse.ccall_preserve_uses)
753 nuses = 0
754 for idx in intermediaries
755 nuses += used_ssas[idx]
756 end
757 nuses_total = used_ssas[idx] + nuses - length(intermediaries)
758 nleaves == nuses_total || continue
759 # Find the type for this allocation
760 defexpr = ir[SSAValue(idx)]
761 isexpr(defexpr, :new) || continue
762 typ = ir.stmts[idx][:type]
763 if isa(typ, UnionAll)
764 typ = unwrap_unionall(typ)
765 end
766 # Could still end up here if we tried to setfield! and immutable, which would
767 # error at runtime, but is not illegal to have in the IR.
768 ismutabletype(typ) || continue
769 # Partition defuses by field
770 fielddefuse = SSADefUse[SSADefUse() for _ = 1:fieldcount(typ)]
771 ok = true
772 for use in defuse.uses
773 stmt = ir[SSAValue(use)]
774 # We may have discovered above that this use is dead
775 # after the getfield elim of immutables. In that case,
776 # it would have been deleted. That's fine, just ignore
777 # the use in that case.
778 stmt === nothing && continue
779 field = try_compute_fieldidx_expr(typ, stmt)
780 field === nothing && (ok = false; break)
781 push!(fielddefuse[field].uses, use)
782 end
783 ok || continue
784 for use in defuse.defs
785 field = try_compute_fieldidx_expr(typ, ir[SSAValue(use)])
786 field === nothing && (ok = false; break)
787 push!(fielddefuse[field].defs, use)
788 end
789 ok || continue
790 # Check that the defexpr has defined values for all the fields
791 # we're accessing. In the future, we may want to relax this,
792 # but we should come up with semantics for well defined semantics
793 # for uninitialized fields first.
794 for (fidx, du) in pairs(fielddefuse)
795 isempty(du.uses) && continue
796 if fidx + 1 > length(defexpr.args)
797 ok = false
798 break
799 end
800 end
801 ok || continue
802 preserve_uses = IdDict{Int, Vector{Any}}((idx=>Any[] for idx in IdSet{Int}(defuse.ccall_preserve_uses)))
803 # Everything accounted for. Go field by field and perform idf
804 for (fidx, du) in pairs(fielddefuse)
805 ftyp = fieldtype(typ, fidx)
806 if !isempty(du.uses)
807 push!(du.defs, idx)
808 ldu = compute_live_ins(ir.cfg, du)
809 phiblocks = Int[]
810 if !isempty(ldu.live_in_bbs)
811 phiblocks = idf(ir.cfg, ldu, domtree)
812 end
813 phinodes = IdDict{Int, SSAValue}()
814 for b in phiblocks
815 n = PhiNode()
816 phinodes[b] = insert_node!(ir, first(ir.cfg.blocks[b].stmts),
817 NewInstruction(n, ftyp))
818 end
819 # Now go through all uses and rewrite them
820 allblocks = sort(vcat(phiblocks, ldu.def_bbs))
821 for stmt in du.uses
822 ir[SSAValue(stmt)] = compute_value_for_use(ir, domtree, allblocks, du, phinodes, fidx, stmt)
823 end
824 if !isbitstype(ftyp)
825 for (use, list) in preserve_uses
826 push!(list, compute_value_for_use(ir, domtree, allblocks, du, phinodes, fidx, use))
827 end
828 end
829 for b in phiblocks
830 for p in ir.cfg.blocks[b].preds
831 n = ir[phinodes[b]]
832 push!(n.edges, p)
833 push!(n.values, compute_value_for_block(ir, domtree,
834 allblocks, du, phinodes, fidx, p))
835 end
836 end
837 end
838 for stmt in du.defs
839 stmt == idx && continue
840 ir[SSAValue(stmt)] = nothing
841 end
842 continue
843 end
844 isempty(defuse.ccall_preserve_uses) && continue
845 push!(intermediaries, idx)
846 # Insert the new preserves
847 for (use, new_preserves) in preserve_uses
848 useexpr = ir[SSAValue(use)]
849 nccallargs = length(useexpr.args[3]::SimpleVector)
850 old_preserves = let intermediaries = intermediaries
851 filter(ssa->!isa(ssa, SSAValue) || !(ssa.id in intermediaries), useexpr.args[(6+nccallargs):end])
852 end
853 new_expr = Expr(:foreigncall, useexpr.args[1:(6+nccallargs-1)]...,
854 old_preserves..., new_preserves...)
855 ir[SSAValue(use)] = new_expr
856 end
857 end
858 ir
859 end

895 function adce_pass!(ir::IRCode)
896 phi_uses = fill(0, length(ir.stmts) + length(ir.new_nodes))
897 all_phis = Int[]
898 compact = IncrementalCompact(ir)
899 for ((_, idx), stmt) in compact
900 if isa(stmt, PhiNode)
901 push!(all_phis, idx)
902 end
903 end
904 non_dce_finish!(compact)
905 for phi in all_phis
906 count_uses(compact.result[phi][:inst]::PhiNode, phi_uses)
907 end
908 # Perform simple DCE for unused values
909 extra_worklist = Int[]
910 for (idx, nused) in Iterators.enumerate(compact.used_ssas)
911 idx >= compact.result_idx && break
912 nused == 0 || continue
913 adce_erase!(phi_uses, extra_worklist, compact, idx)
914 end
915 while !isempty(extra_worklist)
916 adce_erase!(phi_uses, extra_worklist, compact, pop!(extra_worklist))
917 end
918 # Go back and erase any phi cycles
919 changed = true
920 while changed
921 changed = false
922 safe_phis = BitSet()
923 for phi in all_phis
924 # Save any phi cycles that have non-phi uses
925 if compact.used_ssas[phi] - phi_uses[phi] != 0
926 mark_phi_cycles(compact, safe_phis, phi)
927 end
928 end
929 for phi in all_phis
930 if !(phi in safe_phis)
931 push!(extra_worklist, phi)
932 end
933 end
934 while !isempty(extra_worklist)
935 if adce_erase!(phi_uses, extra_worklist, compact, pop!(extra_worklist))
936 changed = true
937 end
938 end
939 end
940 return complete(compact)
941 end

972 val = def.val

1010 end

1044 end
1045 else

1047 end
1048 end

1078 function cfg_simplify!(ir::IRCode)
1079 bbs = ir.cfg.blocks
1080 merge_into = zeros(Int, length(bbs))
1081 merged_succ = zeros(Int, length(bbs))
1082 function follow_merge_into(idx::Int)
1083 while merge_into[idx] != 0
1084 idx = merge_into[idx]
1085 end
1086 return idx
1087 end
1088 function follow_merged_succ(idx::Int)
1089 while merged_succ[idx] != 0
1090 idx = merged_succ[idx]
1091 end
1092 return idx
1093 end
1094
1095 # Walk the CFG from the entry block and aggressively combine blocks
1096 for (idx, bb) in enumerate(bbs)
1097 if length(bb.succs) == 1
1098 succ = bb.succs[1]
1099 if length(bbs[succ].preds) == 1
1100 # Prevent cycles by making sure we don't end up back at `idx`
1101 # by following what is to be merged into `succ`
1102 if follow_merged_succ(succ) != idx
1103 merge_into[succ] = idx
1104 merged_succ[idx] = succ
1105 end
1106 end
1107 end
1108 end
1109
1110 # Assign new BB numbers
1111 max_bb_num = 1
1112 bb_rename_succ = zeros(Int, length(bbs))
1113 for i = 1:length(bbs)
1114 # Drop blocks that will be merged away
1115 if merge_into[i] != 0
1116 bb_rename_succ[i] = -1
1117 end
1118 # Drop blocks with no predecessors
1119 if i != 1 && length(ir.cfg.blocks[i].preds) == 0
1120 bb_rename_succ[i] = -1
1121 end
1122
1123 bb_rename_succ[i] != 0 && continue
1124
1125 curr = i
1126 while true
1127 bb_rename_succ[curr] = max_bb_num
1128 max_bb_num += 1
1129 # Now walk the chain of blocks we merged.
1130 # If we end in something that may fall through,
1131 # we have to schedule that block next
1132 curr = follow_merged_succ(curr)
1133 terminator = ir.stmts[ir.cfg.blocks[curr].stmts[end]][:inst]
1134 if isa(terminator, GotoNode) || isa(terminator, ReturnNode)
1135 break
1136 end
1137 curr += 1
1138 end
1139 end
1140
1141 # Figure out how predecessors should be renamed
1142 bb_rename_pred = zeros(Int, length(bbs))
1143 for i = 1:length(bbs)
1144 if merged_succ[i] != 0
1145 # Block `i` should no longer be a predecessor (before renaming)
1146 # because it is being merged with its sole successor
1147 bb_rename_pred[i] = -1
1148 continue
1149 end
1150 bbnum = follow_merge_into(i)
1151 bb_rename_pred[i] = bb_rename_succ[bbnum]
1152 end
1153
1154 # Compute map from new to old blocks
1155 result_bbs = Int[findfirst(j->i==j, bb_rename_succ) for i = 1:max_bb_num-1]
1156
1157 # Compute new block lengths
1158 result_bbs_lengths = zeros(Int, max_bb_num-1)
1159 for (idx, orig_bb) in enumerate(result_bbs)
1160 ms = orig_bb
1161 while ms != 0
1162 result_bbs_lengths[idx] += length(bbs[ms].stmts)
1163 ms = merged_succ[ms]
1164 end
1165 end
1166
1167 # Compute statement indices the new blocks start at
1168 bb_starts = Vector{Int}(undef, 1+length(result_bbs_lengths))
1169 bb_starts[1] = 1
1170 for i = 1:length(result_bbs_lengths)
1171 bb_starts[i+1] = bb_starts[i] + result_bbs_lengths[i]
1172 end
1173
1174 cresult_bbs = let result_bbs = result_bbs,
1175 merged_succ = merged_succ,
1176 merge_into = merge_into,
1177 bbs = bbs,
1178 bb_rename_succ = bb_rename_succ
1179
1180 # Compute (renamed) successors and predecessors given (renamed) block
1181 function compute_succs(i)
1182 orig_bb = follow_merged_succ(result_bbs[i])
1183 return Int[bb_rename_succ[i] for i in bbs[orig_bb].succs]
1184 end
1185 function compute_preds(i)
1186 orig_bb = result_bbs[i]
1187 preds = bbs[orig_bb].preds
1188 return Int[bb_rename_pred[pred] for pred in preds]
1189 end
1190
1191 BasicBlock[
1192 BasicBlock(StmtRange(bb_starts[i],
1193 i+1 > length(bb_starts) ?
1194 length(compact.result) : bb_starts[i+1]-1),
1195 compute_preds(i),
1196 compute_succs(i))
1197 for i = 1:length(result_bbs)]
1198 end
1199
1200 compact = IncrementalCompact(ir, true)
1201 # Run instruction compaction to produce the result,
1202 # but we're messing with the CFG
1203 # so we don't want compaction to do so independently
1204 compact.fold_constant_branches = false
1205 compact.bb_rename_succ = bb_rename_succ
1206 compact.bb_rename_pred = bb_rename_pred
1207 compact.result_bbs = cresult_bbs
1208 result_idx = 1
1209 for (idx, orig_bb) in enumerate(result_bbs)
1210 ms = orig_bb
1211 while ms != 0
1212 for i in bbs[ms].stmts
1213 node = ir.stmts[i]
1214 compact.result[compact.result_idx] = node
1215 if isa(node[:inst], GotoNode) && merged_succ[ms] != 0
1216 # If we merged a basic block, we need remove the trailing GotoNode (if any)
1217 compact.result[compact.result_idx][:inst] = nothing
1218 else
1219 process_node!(compact, compact.result_idx, node, i, i, ms, true)
1220 end
1221 # We always increase the result index to ensure a predicatable
1222 # placement of the resulting nodes.
1223 compact.result_idx += 1
1224 end
1225 ms = merged_succ[ms]
1226 end
1227 end
1228 compact.active_result_bb = length(bb_starts)
1229 return finish(compact)
1230 end
