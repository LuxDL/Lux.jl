38 """
39 AsyncCondition(callback::Function)
40
41 Create a async condition that calls the given `callback` function. The `callback` is passed one argument,
42 the async condition object itself.
43 """
44 function AsyncCondition(cb::Function)
45 async = AsyncCondition()
46 t = @task while _trywait(async)
47 cb(async)
48 isopen(async) || return
49 end
50 lock(async.cond)
51 _wait2(async.cond, t)
52 unlock(async.cond)
53 return async
54 end

58 """
59 Timer(delay; interval = 0)
60
61 Create a timer that wakes up tasks waiting for it (by calling [`wait`](@ref) on the timer object).
62
63 Waiting tasks are woken after an initial delay of at least `delay` seconds, and then repeating after
64 at least `interval` seconds again elapse. If `interval` is equal to `0`, the timer is only triggered
65 once. When the timer is closed (by [`close`](@ref)) waiting tasks are woken with an error. Use
66 [`isopen`](@ref) to check whether a timer is still active.
67
68 Note: `interval` is subject to accumulating time skew. If you need precise events at a particular
69 absolute time, create a new timer at each expiration with the difference to the next time computed.
70 """
71 mutable struct Timer
72 handle::Ptr{Cvoid}
73 cond::ThreadSynchronizer
74 isopen::Bool
75 set::Bool
76
77 function Timer(timeout::Real; interval::Real = 0.0)
78 timeout ≥ 0 || throw(ArgumentError("timer cannot have negative timeout of $timeout seconds"))
79 interval ≥ 0 || throw(ArgumentError("timer cannot have negative repeat interval of $interval seconds"))
80 # libuv has a tendency to timeout 1 ms early, so we need +1 on the timeout (in milliseconds), unless it is zero
81 timeoutms = ceil(UInt64, timeout * 1000) + !iszero(timeout)
82 intervalms = ceil(UInt64, interval * 1000)
83 loop = eventloop()
84
85 this = new(Libc.malloc(_sizeof_uv_timer), ThreadSynchronizer(), true, false)
86 associate_julia_struct(this.handle, this)
87 iolock_begin()
88 err = ccall(:uv_timer_init, Cint, (Ptr{Cvoid}, Ptr{Cvoid}), loop, this)
89 @assert err == 0
90 finalizer(uvfinalize, this)
91 ccall(:uv_update_time, Cvoid, (Ptr{Cvoid},), loop)
92 err = ccall(:uv_timer_start, Cint, (Ptr{Cvoid}, Ptr{Cvoid}, UInt64, UInt64),
93 this, @cfunction(uv_timercb, Cvoid, (Ptr{Cvoid},)),
94 timeoutms, intervalms)
95 @assert err == 0
96 iolock_end()
97 return this
98 end
99 end

123 end

225 # timer with repeated callback
226 """
227 Timer(callback::Function, delay; interval = 0)
228
229 Create a timer that runs the function `callback` at each timer expiration.
230
231 Waiting tasks are woken and the function `callback` is called after an initial delay of `delay`
232 seconds, and then repeating with the given `interval` in seconds. If `interval` is equal to `0`, the
233 callback is only run once. The function `callback` is called with a single argument, the timer
234 itself. Stop a timer by calling `close`. The `cb` may still be run one final time, if the timer has
235 already expired.
236
237 # Examples
238
239 Here the first number is printed after a delay of two seconds, then the following numbers are
240 printed quickly.
241
242 ```julia-repl
243 julia> begin
244 i = 0
245 cb(timer) = (global i += 1; println(i))
246 t = Timer(cb, 2, interval=0.2)
247 wait(t)
248 sleep(0.5)
249 close(t)
250 end
251 1
252 2
253 3
254 ```
255 """
256 function Timer(cb::Function, timeout::Real; interval::Real=0.0)
257 timer = Timer(timeout, interval=interval)
258 t = @task while _trywait(timer)
259 try
260 cb(timer)
261 catch err
262 write(stderr, "Error in Timer:\n")
263 showerror(stderr, err, catch_backtrace())
264 return
265 end
266 isopen(timer) || return
267 end
268 lock(timer.cond)
269 _wait2(timer.cond, t)
270 unlock(timer.cond)
271 return timer
272 end

274 """
275 timedwait(callback::Function, timeout::Real; pollint::Real=0.1)
276
277 Waits until `callback` returns `true` or `timeout` seconds have passed, whichever is earlier.
278 `callback` is polled every `pollint` seconds. The minimum value for `timeout` and `pollint`
279 is `0.001`, that is, 1 millisecond.
280
281 Returns :ok or :timed_out
282 """
283 function timedwait(testcb::Function, timeout::Real; pollint::Real=0.1)
284 pollint >= 1e-3 || throw(ArgumentError("pollint must be ≥ 1 millisecond"))
285 start = time_ns()
286 ns_timeout = 1e9 * timeout
287 done = Channel(1)
288 function timercb(aw)
289 try
290 if testcb()
291 put!(done, (:ok, nothing))
292 elseif (time_ns() - start) > ns_timeout
293 put!(done, (:timed_out, nothing))
294 end
295 catch e
296 put!(done, (:error, CapturedException(e, catch_backtrace())))
297 finally
298 isready(done) && close(aw)
299 end
300 nothing
301 end
302
303 try
304 testcb() && return :ok
305 catch e
306 throw(CapturedException(e, catch_backtrace()))
307 end
308
309 t = Timer(timercb, pollint, interval = pollint)
310 ret, e = fetch(done)
311 close(t)
312
313 ret === :error && throw(e)
314
315 return ret
316 end
