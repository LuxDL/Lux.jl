51 # A dict of all libuv handles that are being waited on somewhere in the system
52 # and should thus not be garbage collected
53 const uvhandles = IdDict()
54 const preserve_handle_lock = Threads.SpinLock()
55 function preserve_handle(x)
56 lock(preserve_handle_lock)
57 v = get(uvhandles, x, 0)::Int
58 uvhandles[x] = v + 1
59 unlock(preserve_handle_lock)
60 nothing
61 end
62 function unpreserve_handle(x)
63 lock(preserve_handle_lock)
64 v = uvhandles[x]::Int
65 if v == 1
66 pop!(uvhandles, x)
67 else
68 uvhandles[x] = v - 1
69 end
70 unlock(preserve_handle_lock)
71 nothing
72 end
