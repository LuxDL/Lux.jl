3 function _truncate_at_width_or_chars(str, width, chars="", truncmark="…")
4 truncwidth = textwidth(truncmark)
5 (width <= 0 || width < truncwidth) && return ""
6
7 wid = truncidx = lastidx = 0
8 for (idx, c) in pairs(str)
9 lastidx = idx
10 wid += textwidth(c)
11 wid >= width - truncwidth && truncidx == 0 && (truncidx = lastidx)
12 (wid >= width || c in chars) && break
13 end
14
15 lastidx != 0 && str[lastidx] in chars && (lastidx = prevind(str, lastidx))
16 truncidx == 0 && (truncidx = lastidx)
17 if lastidx < lastindex(str)
18 return String(SubString(str, 1, truncidx) * truncmark)
19 else
20 return String(str)
21 end
22 end

176 olds = h.slots

178 oldv = h.vals
179 sz = length(olds)
180 newsz = _tablesz(newsz)
181 h.age += 1
182 h.idxfloor = 1
183 if h.count == 0
184 resize!(h.slots, newsz)
185 fill!(h.slots, 0)
186 resize!(h.keys, newsz)
187 resize!(h.vals, newsz)
188 h.ndel = 0
189 return h
190 end

192 slots = zeros(UInt8,newsz)
193 keys = Vector{K}(undef, newsz)
194 vals = Vector{V}(undef, newsz)
195 age0 = h.age
196 count = 0
197 maxprobe = 0
198
199 for i = 1:sz
200 @inbounds if olds[i] == 0x1
201 k = oldk[i]
202 v = oldv[i]
203 index0 = index = hashindex(k, newsz)
204 while slots[index] != 0
205 index = (index & (newsz-1)) + 1
206 end
207 probe = (index - index0) & (newsz-1)

211 vals[index] = v

213

215 # if `h` is changed by a finalizer, retry
216 return rehash!(h, newsz)
217 end
218 end
219 end

222 h.keys = keys

758 `ImmutableDict` is a dictionary implemented as an immutable linked list,
759 which is optimal for small dictionaries that are constructed over many individual insertions.
760 Note that it is not possible to remove a value, although it can be partially overridden and hidden
761 by inserting a new value with the same key.
762
763 ImmutableDict(KV::Pair)
764
765 Create a new entry in the `ImmutableDict` for a `key => value` pair
766
767 - use `(key => value) in dict` to see if this particular combination is in the properties set
768 - use `get(dict, key, default)` to retrieve the most recent value for a particular key
769

778 function in(key_value::Pair, dict::ImmutableDict, valcmp=(==))
779 key, value = key_value
780 while isdefined(dict, :parent)
781 if isequal(dict.key, key)
782 valcmp(value, dict.value) && return true
783 end
784 dict = dict.parent
785 end
786 return false
787 end
788
789 function haskey(dict::ImmutableDict, key)
790 while isdefined(dict, :parent)
791 isequal(dict.key, key) && return true
792 dict = dict.parent
793 end
794 return false
795 end
796
797 function getindex(dict::ImmutableDict, key)
798 while isdefined(dict, :parent)
799 isequal(dict.key, key) && return dict.value
800 dict = dict.parent
801 end
802 throw(KeyError(key))
803 end
804 function get(dict::ImmutableDict, key, default)
805 while isdefined(dict, :parent)
806 isequal(dict.key, key) && return dict.value
807 dict = dict.parent
808 end
809 return default
810 end

812 function get(default::Callable, dict::ImmutableDict, key)
813 while isdefined(dict, :parent)
814 isequal(dict.key, key) && return dict.value
815 dict = dict.parent
816 end
817 return default()
818 end
