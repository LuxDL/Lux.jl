8 """
9 @cuda [kwargs...] func(args...)
10
11 High-level interface for executing code on a GPU. The `@cuda` macro should prefix a call,
12 with `func` a callable function or object that should return nothing. It will be compiled to
13 a CUDA function upon first use, and to a certain extent arguments will be converted and
14 managed automatically using `cudaconvert`. Finally, a call to `cudacall` is
15 performed, scheduling a kernel launch on the current CUDA context.
16
17 Several keyword arguments are supported that influence the behavior of `@cuda`.
18 - `launch`: whether to launch this kernel, defaults to `true`. If `false` the returned
19 kernel object should be launched by calling it and passing arguments again.
20 - `dynamic`: use dynamic parallelism to launch device-side kernels, defaults to `false`.
21 - arguments that influence kernel compilation: see [`cufunction`](@ref) and
22 [`dynamic_cufunction`](@ref)

24 [`CUDA.DeviceKernel`](@ref)

180
181 # filter out arguments that shouldn't be passed
182 predicate = dt -> isghosttype(dt) || Core.Compiler.isconstType(dt)
183 to_pass = map(!predicate, sig.parameters)
184 call_t = Type[x[1] for x in zip(sig.parameters, to_pass) if x[2]]
185 call_args = Union{Expr,Symbol}[x[1] for x in zip(args, to_pass) if x[2]]

193 end
194 end

271 """
272 cufunction(f, tt=Tuple{}; kwargs...)
273
274 Low-level interface to compile a function invocation for the currently-active GPU, returning
275 a callable kernel object. For a higher-level interface, use [`@cuda`](@ref).
276
277 The following keyword arguments are supported:
278 - `minthreads`: the required number of threads in a thread block
279 - `maxthreads`: the maximum number of threads in a thread block
280 - `blocks_per_sm`: a minimum number of thread blocks to be scheduled on a single
281 multiprocessor
282 - `maxregs`: the maximum number of registers to be allocated to a single thread (only
283 supported on LLVM 4.0+)
284 - `name`: override the name that the kernel will have in the generated code
285
286 The output of this function is automatically cached, i.e. you can simply call `cufunction`
287 in a hot path without degrading performance. New code will be generated automatically, when
288 when function changes, or when different types or keyword arguments are provided.
289 """
290 @timeit_ci function cufunction(f::F, tt::TT=Tuple{}; name=nothing, kwargs...) where {F,TT}
291 cuda = active_state()
292 cache = cufunction_cache(cuda.context)
293 source = FunctionSpec(f, tt, true, name)
294 target = CUDACompilerTarget(cuda.device; kwargs...)
295 params = CUDACompilerParams()
296 job = CompilerJob(target, source, params)
297 return GPUCompiler.cached_compilation(cache, job,
298 cufunction_compile,
299 cufunction_link)::HostKernel{F,tt}
300 end

328 @timeit_ci "compile" function cufunction_compile(@nospecialize(job::CompilerJob), ctx::Context)
329 # lower to PTX
330 mi, mi_meta = @timeit_ci "emit_julia" GPUCompiler.emit_julia(job)
331 ir, ir_meta = @timeit_ci "emit_llvm" GPUCompiler.emit_llvm(job, mi; ctx)
332 asm, asm_meta = @timeit_ci "emit_asm" GPUCompiler.emit_asm(job, ir; format=LLVM.API.LLVMAssemblyFile)
333
334 # remove extraneous debug info on lower debug levels
335 if Base.JLOptions().debug_level < 2
336 # LLVM sets `.target debug` as soon as the debug emission kind isn't NoDebug. this
337 # is unwanted, as the flag makes `ptxas` behave as if `--device-debug` were set.
338 # ideally, we'd need something like LocTrackingOnly/EmitDebugInfo from D4234, but
339 # that got removed in favor of NoDebug in D18808, seemingly breaking the use case of
340 # only emitting `.loc` instructions...
341 #
342 # according to NVIDIA, "it is fine for PTX producers to produce debug info but not
343 # set `.target debug` and if `--device-debug` isn't passed, PTXAS will compile in
344 # release mode".
345 asm = replace(asm, r"(\.target .+), debug" => s"\1")
346 end
347
348 # check if we'll need the device runtime
349 undefined_fs = filter(collect(functions(ir))) do f
350 isdeclaration(f) && !LLVM.isintrinsic(f)
351 end
352 intrinsic_fns = ["vprintf", "malloc", "free", "__assertfail",
353 "__nvvm_reflect" #= TODO: should have been optimized away =#]
354 needs_cudadevrt = !isempty(setdiff(LLVM.name.(undefined_fs), intrinsic_fns))
355
356 # find externally-initialized global variables; we'll access those using CUDA APIs.
357 external_gvars = filter(isextinit, collect(globals(ir))) .|> LLVM.name
358
359 # prepare invocations of CUDA compiler tools
360 ptxas_opts = String[]
361 nvlink_opts = String[]
362 ## debug flags
363 if Base.JLOptions().debug_level == 1
364 push!(ptxas_opts, "--generate-line-info")
365 elseif Base.JLOptions().debug_level >= 2
366 push!(ptxas_opts, "--device-debug")
367 push!(nvlink_opts, "--debug")
368 end
369 ## relocatable device code
370 if needs_cudadevrt
371 push!(ptxas_opts, "--compile-only")
372 end
373
374 arch = "sm_$(job.target.cap.major)$(job.target.cap.minor)"
375
376 # compile to machine code
377 # NOTE: we use tempname since mktemp doesn't support suffixes, and mktempdir is slow
378 ptx_input = tempname(cleanup=false) * ".ptx"
379 ptxas_output = tempname(cleanup=false) * ".cubin"
380 write(ptx_input, asm)
381
382 # we could use the driver's embedded JIT compiler, but that has several disadvantages:
383 # 1. fixes and improvements are slower to arrive, by using `ptxas` we only need to
384 # upgrade the toolkit to get a newer compiler;
385 # 2. version checking is simpler, we otherwise need to use NVML to query the driver
386 # version, which is hard to correlate to PTX JIT improvements;
387 # 3. if we want to be able to use newer (minor upgrades) of the CUDA toolkit on an
388 # older driver, we should use the newer compiler to ensure compatibility.
389 append!(ptxas_opts, [
390 "--verbose",
391 "--gpu-name", arch,
392 "--output-file", ptxas_output,
393 ptx_input
394 ])
395 proc, log = @timeit_ci "ptxas" run_and_collect(`$(ptxas()) $ptxas_opts`)
396 log = strip(log)
397 if !success(proc)
398 reason = proc.termsignal > 0 ? "ptxas received signal $(proc.termsignal)" :
399 "ptxas exited with code $(proc.exitcode)"
400 msg = "Failed to compile PTX code ($reason)"
401 if !isempty(log)
402 msg *= "\n" * log
403 end
404 msg *= "\nIf you think this is a bug, please file an issue and attach $(ptx_input)"
405 error(msg)
406 elseif !isempty(log)
407 @debug "PTX compiler log:\n" * log
408 end
409 rm(ptx_input)
410
411 # link device libraries, if necessary
412 #
413 # this requires relocatable device code, which prevents certain optimizations and
414 # hurts performance. as such, we only do so when absolutely necessary.
415 # TODO: try LTO, `--link-time-opt --nvvmpath /opt/cuda/nvvm`.
416 # fails with `Ignoring -lto option because no LTO objects found`
417 if needs_cudadevrt
418 nvlink_output = tempname(cleanup=false) * ".cubin"
419 append!(nvlink_opts, [
420 "--verbose", "--extra-warnings",
421 "--arch", arch,
422 "--library-path", dirname(libcudadevrt()),
423 "--library", "cudadevrt",
424 "--output-file", nvlink_output,
425 ptxas_output
426 ])
427 proc, log = @timeit_ci "nvlink" run_and_collect(`$(nvlink()) $nvlink_opts`)
428 log = strip(log)
429 if !success(proc)
430 reason = proc.termsignal > 0 ? "nvlink received signal $(proc.termsignal)" :
431 "nvlink exited with code $(proc.exitcode)"
432 msg = "Failed to link PTX code ($reason)"
433 if !isempty(log)
434 msg *= "\n" * log
435 end
436 msg *= "\nIf you think this is a bug, please file an issue and attach $(ptxas_output)"
437 error(msg)
438 elseif !isempty(log)
439 @debug "PTX linker info log:\n" * log
440 end
441 rm(ptxas_output)
442
443 image = read(nvlink_output)
444 rm(nvlink_output)
445 else
446 image = read(ptxas_output)
447 rm(ptxas_output)
448 end
449
450 return (image, entry=LLVM.name(ir_meta.entry), external_gvars)
451 end

505 nextwarp(dev, threads)
506 prevwarp(dev, threads)
507
508 Returns the next or previous nearest number of threads that is a multiple of the warp size
509 of a device `dev`. This is a common requirement when using intra-warp communication.
510 """
511 function nextwarp(dev::CuDevice, threads::Integer)
512 ws = warpsize(dev)
513 return threads + (ws - threads % ws) % ws
514 end

516 @doc (@doc nextwarp) function prevwarp(dev::CuDevice, threads::Integer)
517 ws = warpsize(dev)
518 return threads - Base.rem(threads, ws)
