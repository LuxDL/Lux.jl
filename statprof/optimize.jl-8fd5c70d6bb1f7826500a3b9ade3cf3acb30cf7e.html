159 function isinlineable(m::Method, me::OptimizationState, params::OptimizationParams, union_penalties::Bool, bonus::Int=0)
160 # compute the cost (size) of inlining this code
161 inlineable = false
162 cost_threshold = params.inline_cost_threshold
163 if m.module === _topmod(m.module)
164 # a few functions get special treatment
165 name = m.name
166 sig = m.sig
167 if ((name === :+ || name === :* || name === :min || name === :max) &&
168 isa(sig,DataType) &&
169 sig == Tuple{sig.parameters[1],Any,Any,Any,Vararg{Any}})
170 inlineable = true
171 elseif (name === :iterate || name === :unsafe_convert ||
172 name === :cconvert)
173 cost_threshold *= 4
174 end
175 end
176 if !inlineable
177 inlineable = inline_worthy(me.ir, params, union_penalties, cost_threshold + bonus)
178 end
179 return inlineable
180 end

445 function statement_cost(ex::Expr, line::Int, src::Union{CodeInfo, IRCode}, sptypes::Vector{Any},
446 slottypes::Vector{Any}, union_penalties::Bool,
447 params::OptimizationParams, error_path::Bool = false)
448 head = ex.head
449 if is_meta_expr_head(head)
450 return 0
451 elseif head === :call
452 farg = ex.args[1]
453 ftyp = argextype(farg, src, sptypes, slottypes)
454 if ftyp === IntrinsicFunction && farg isa SSAValue
455 # if this comes from code that was already inlined into another function,
456 # Consts have been widened. try to recover in simple cases.
457 farg = isa(src, CodeInfo) ? src.code[farg.id] : src.stmts[farg.id][:inst]
458 if isa(farg, GlobalRef) || isa(farg, QuoteNode) || isa(farg, IntrinsicFunction) || isexpr(farg, :static_parameter)
459 ftyp = argextype(farg, src, sptypes, slottypes)
460 end
461 end
462 f = singleton_type(ftyp)
463 if isa(f, IntrinsicFunction)
464 iidx = Int(reinterpret(Int32, f::IntrinsicFunction)) + 1
465 if !isassigned(T_IFUNC_COST, iidx)
466 # unknown/unhandled intrinsic
467 return params.inline_nonleaf_penalty
468 end
469 return T_IFUNC_COST[iidx]
470 end
471 if isa(f, Builtin)
472 # The efficiency of operations like a[i] and s.b
473 # depend strongly on whether the result can be
474 # inferred, so check the type of ex
475 if f === Core.getfield || f === Core.tuple
476 # we might like to penalize non-inferrability, but
477 # tuple iteration/destructuring makes that impossible
478 # return plus_saturate(argcost, isknowntype(extyp) ? 1 : params.inline_nonleaf_penalty)
479 return 0
480 elseif (f === Core.arrayref || f === Core.const_arrayref || f === Core.arrayset) && length(ex.args) >= 3
481 atyp = argextype(ex.args[3], src, sptypes, slottypes)
482 return isknowntype(atyp) ? 4 : error_path ? params.inline_error_path_cost : params.inline_nonleaf_penalty
483 elseif f === typeassert && isconstType(widenconst(argextype(ex.args[3], src, sptypes, slottypes)))
484 return 1
485 elseif f === Core.isa
486 # If we're in a union context, we penalize type computations
487 # on union types. In such cases, it is usually better to perform
488 # union splitting on the outside.
489 if union_penalties && isa(argextype(ex.args[2], src, sptypes, slottypes), Union)
490 return params.inline_nonleaf_penalty
491 end
492 end
493 fidx = find_tfunc(f)
494 if fidx === nothing
495 # unknown/unhandled builtin
496 # Use the generic cost of a direct function call
497 return 20
498 end
499 return T_FFUNC_COST[fidx]
500 end
501 extyp = line == -1 ? Any : argextype(SSAValue(line), src, sptypes, slottypes)
502 if extyp === Union{}
503 return 0
504 end
505 return error_path ? params.inline_error_path_cost : params.inline_nonleaf_penalty
506 elseif head === :foreigncall || head === :invoke || head == :invoke_modify
507 # Calls whose "return type" is Union{} do not actually return:
508 # they are errors. Since these are not part of the typical
509 # run-time of the function, we omit them from
510 # consideration. This way, non-inlined error branches do not
511 # prevent inlining.
512 extyp = line == -1 ? Any : argextype(SSAValue(line), src, sptypes, slottypes)
513 return extyp === Union{} ? 0 : 20
514 elseif head === :(=)
515 if ex.args[1] isa GlobalRef
516 cost = 20
517 else
518 cost = 0
519 end
520 a = ex.args[2]
521 if a isa Expr
522 cost = plus_saturate(cost, statement_cost(a, -1, src, sptypes, slottypes, union_penalties, params, error_path))
523 end
524 return cost
525 elseif head === :copyast
526 return 100
527 elseif head === :enter
528 # try/catch is a couple function calls,
529 # but don't inline functions with try/catch
530 # since these aren't usually performance-sensitive functions,
531 # and llvm is more likely to miscompile them when these functions get large
532 return typemax(Int)
533 end
534 return 0
535 end

537 function statement_or_branch_cost(@nospecialize(stmt), line::Int, src::Union{CodeInfo, IRCode}, sptypes::Vector{Any},
538 slottypes::Vector{Any}, union_penalties::Bool, params::OptimizationParams,
539 throw_blocks::Union{Nothing,BitSet})
540 thiscost = 0
541 dst(tgt) = isa(src, IRCode) ? first(src.cfg.blocks[tgt].stmts) : tgt
542 if stmt isa Expr
543 thiscost = statement_cost(stmt, line, src, sptypes, slottypes, union_penalties, params,
544 throw_blocks !== nothing && line in throw_blocks)::Int
545 elseif stmt isa GotoNode
546 # loops are generally always expensive
547 # but assume that forward jumps are already counted for from
548 # summing the cost of the not-taken branch
549 thiscost = dst(stmt.label) < line ? 40 : 0
550 elseif stmt isa GotoIfNot
551 thiscost = dst(stmt.dest) < line ? 40 : 0
552 end
553 return thiscost
554 end

556 function inline_worthy(ir::IRCode,
557 params::OptimizationParams, union_penalties::Bool=false, cost_threshold::Integer=params.inline_cost_threshold)
558 bodycost::Int = 0
559 throw_blocks = params.unoptimize_throw_blocks ? find_throw_blocks(ir.stmts.inst, RefValue(ir)) : nothing

561 stmt = ir.stmts[line][:inst]
562 thiscost = statement_or_branch_cost(stmt, line, ir, ir.sptypes, ir.argtypes, union_penalties, params, throw_blocks)
563 bodycost = plus_saturate(bodycost, thiscost)

610 function renumber_ir_elements!(body::Vector{Any}, ssachangemap::Vector{Int}, labelchangemap::Vector{Int})
611 cumsum_ssamap!(labelchangemap)
612 if ssachangemap !== labelchangemap
613 cumsum_ssamap!(ssachangemap)
614 end
615 if labelchangemap[end] == 0 && ssachangemap[end] == 0
616 return
617 end
618 for i = 1:length(body)
619 el = body[i]
620 if isa(el, GotoNode)
621 body[i] = GotoNode(el.label + labelchangemap[el.label])
622 elseif isa(el, GotoIfNot)
623 cond = el.cond
624 if isa(cond, SSAValue)
625 cond = SSAValue(cond.id + ssachangemap[cond.id])
626 end
627 body[i] = GotoIfNot(cond, el.dest + labelchangemap[el.dest])
628 elseif isa(el, ReturnNode)
629 if isdefined(el, :val) && isa(el.val, SSAValue)
630 body[i] = ReturnNode(SSAValue(el.val.id + ssachangemap[el.val.id]))
631 end
632 elseif isa(el, SSAValue)
633 body[i] = SSAValue(el.id + ssachangemap[el.id])
634 elseif isa(el, PhiNode)
635 i = 1
636 edges = el.edges
637 values = el.values
638 while i <= length(edges)
639 was_deleted = ssachangemap[edges[i]] == typemin(Int)
640 if was_deleted
641 deleteat!(edges, i)
642 deleteat!(values, i)
643 else
644 edges[i] += ssachangemap[edges[i]]
645 val = values[i]
646 if isa(val, SSAValue)
647 values[i] = SSAValue(val.id + ssachangemap[val.id])
648 end
649 i += 1
650 end
651 end
652 elseif isa(el, Expr)
653 if el.head === :(=) && el.args[2] isa Expr
654 el = el.args[2]::Expr
655 end
656 if el.head === :enter
657 tgt = el.args[1]::Int
658 el.args[1] = tgt + labelchangemap[tgt]
659 elseif !is_meta_expr_head(el.head)
660 args = el.args
661 for i = 1:length(args)
662 el = args[i]
663 if isa(el, SSAValue)
664 args[i] = SSAValue(el.id + ssachangemap[el.id])
665 end
666 end
667 end
668 end
669 end
