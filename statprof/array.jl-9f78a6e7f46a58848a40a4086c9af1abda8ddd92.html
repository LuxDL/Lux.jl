55 """
56 CUDA.unsafe_free!(a::CuArray, [stream::CuStream])
57
58 Release the memory of an array for reuse by future allocations. This function is
59 automatically called by the finalizer when an array goes out of scope, but can be called
60 earlier to reduce pressure on the memory allocator.
61
62 By default, the operation is performed on the task-local stream. During task or process
63 finalization however, that stream may be destroyed already, so be sure to specify a safe
64 stream (i.e. `default_stream()`, which will ensure the operation will block on other
65 streams) when calling this function from a finalizer. For simplicity, the `unsafe_finalize!`
66 function does exactly that.
67 """
68 function unsafe_free!(xs::CuArray, stream::CuStream=stream())
69 # this call should only have an effect once, because both the user and the GC can call it
70 if xs.storage === nothing
71 return
72 elseif xs.storage.refcount[] < 0
73 throw(ArgumentError("Cannot free an unmanaged buffer."))
74 end
75
76 refcount = Threads.atomic_add!(xs.storage.refcount, -1)
77 if refcount == 1
78 context!(context(xs); skip_destroyed=true) do
79 free(xs.storage.buffer; stream)
80 end
81 end
82
83 # this array object is now dead, so replace its storage by a dummy one
84 xs.storage = nothing
85
86 return
87 end
88
89 function unsafe_finalize!(xs::CuArray)
90 # during task or process finalization, the local stream might be destroyed already, so
91 # use the default stream. additionally, since we don't use per-thread APIs, this default
92 # stream follows legacy semantics and will synchronize all other streams. this protects
93 # against freeing resources that are still in use.
94 #
95 # TODO: although this is still an asynchronous operation, even when using the default
96 # stream, it synchronizes "too much". we could do better, e.g., by keeping track of all
97 # streams involved, or by refcounting uses and decrementing that refcount after the
98 # operation using `cuLaunchHostFunc`. See CUDA.jl#778 and CUDA.jl#780 for details.
99 unsafe_free!(xs, default_stream())
100 # NOTE: we don't switch contexts here, but in unsafe_free!, as arrays are refcounted
101 # and we may not have to free the memory yet.
102 end

181 unsafe_wrap(::CuArray, ptr::CuPtr{T}, dims; own=false, ctx=context())
182
183 Wrap a `CuArray` object around the data at the address given by `ptr`. The pointer
184 element type `T` determines the array element type. `dims` is either an integer (for a 1d
185 array) or a tuple of the array dimensions. `own` optionally specified whether Julia should
186 take ownership of the memory, calling `cudaFree` when the array is no longer referenced. The
187 `ctx` argument determines the CUDA context where the data is allocated in.
188 """
189 function Base.unsafe_wrap(::Union{Type{CuArray},Type{CuArray{T}},Type{CuArray{T,N}}},
190 ptr::CuPtr{T}, dims::NTuple{N,Int};
191 own::Bool=false, ctx::CuContext=context()) where {T,N}
192 Base.isbitstype(T) || error("Can only unsafe_wrap a pointer to a bits type")
193 sz = prod(dims) * sizeof(T)
194
195 # identify the buffer
196 buf = try
197 typ = memory_type(ptr)
198 if is_managed(ptr)
199 Mem.UnifiedBuffer(ctx, ptr, sz)
200 elseif typ == CU_MEMORYTYPE_DEVICE
201 # TODO: can we identify whether this pointer was allocated asynchronously?
202 Mem.DeviceBuffer(ctx, ptr, sz, false)
203 elseif typ == CU_MEMORYTYPE_HOST
204 Mem.HostBuffer(ctx, host_pointer(ptr), sz)
205 else
206 error("Unknown memory type; please file an issue.")
207 end
208 catch err
209 error("Could not identify the buffer type; are you passing a valid CUDA pointer to unsafe_wrap?")
210 end
211
212 storage = ArrayStorage(buf, own ? 1 : -1)
213 CuArray{T, length(dims)}(storage, dims)
214 end

693 ## reinterpret
694
695 function Base.reinterpret(::Type{T}, a::CuArray{S,N}) where {T,S,N}
696 err = _reinterpret_exception(T, a)
697 err === nothing || throw(err)
698
699 if sizeof(T) == sizeof(S) # for N == 0
700 osize = size(a)
701 else
702 isize = size(a)
703 size1 = div(isize[1]*sizeof(S), sizeof(T))
704 osize = tuple(size1, Base.tail(isize)...)
705 end
706
707 return _derived_array(T, N, a, osize)
708 end
709
710 function _reinterpret_exception(::Type{T}, a::AbstractArray{S,N}) where {T,S,N}
711 if !isbitstype(T) || !isbitstype(S)
712 return _CuReinterpretBitsTypeError{T,typeof(a)}()
713 end
714 if N == 0 && sizeof(T) != sizeof(S)
715 return _CuReinterpretZeroDimError{T,typeof(a)}()
716 end
717 if N != 0 && sizeof(S) != sizeof(T)
718 ax1 = axes(a)[1]
719 dim = length(ax1)
720 if Base.rem(dim*sizeof(S),sizeof(T)) != 0
721 return _CuReinterpretDivisibilityError{T,typeof(a)}(dim)
722 end
723 if first(ax1) != 1
724 return _CuReinterpretFirstIndexError{T,typeof(a),typeof(ax1)}(ax1)
725 end
726 end
727 return nothing
728 end

761 function Base.reinterpret(::typeof(reshape), ::Type{T}, a::CuArray) where {T}

765
766 # taken from reinterpretarray.jl
767 # TODO: move these Base definitions out of the ReinterpretArray struct for reuse
768 function _base_check_reshape_reinterpret(::Type{T}, a::CuArray{S}) where {T,S}
769 isbitstype(T) || throwbits(S, T, T)
770 isbitstype(S) || throwbits(S, T, S)
771 if sizeof(S) == sizeof(T)
772 N = ndims(a)
773 osize = size(a)
774 elseif sizeof(S) > sizeof(T)
775 d, r = divrem(sizeof(S), sizeof(T))
776 r == 0 || throwintmult(S, T)
777 N = ndims(a) + 1
778 osize = (d, size(a)...)
779 else
780 d, r = divrem(sizeof(T), sizeof(S))
781 r == 0 || throwintmult(S, T)
782 N = ndims(a) - 1
783 N > -1 || throwsize0(S, T, "larger")
784 axes(a, 1) == Base.OneTo(sizeof(T) sizeof(S)) || throwsize1(a, T)
785 osize = size(a)[2:end]
786 end
787 return N, osize
788 end

790 @noinline function throwbits(S::Type, T::Type, U::Type)
791 throw(ArgumentError("cannot reinterpret `$(S)` as `$(T)`, type `$(U)` is not a bits type"))
792 end

794 @noinline function throwintmult(S::Type, T::Type)
795 throw(ArgumentError("`reinterpret(reshape, T, a)` requires that one of `sizeof(T)` (got $(sizeof(T))) and `sizeof(eltype(a))` (got $(sizeof(S))) be an integer multiple of the other"))
796 end

798 @noinline function throwsize0(S::Type, T::Type, msg)
799 throw(ArgumentError("cannot reinterpret a zero-dimensional `$(S)` array to `$(T)` which is of a $msg size"))
800 end

802 @noinline function throwsize1(a::AbstractArray, T::Type)
803 throw(ArgumentError("`reinterpret(reshape, $T, a)` where `eltype(a)` is $(eltype(a)) requires that `axes(a, 1)` (got $(axes(a, 1))) be equal to 1:$(sizeof(T) sizeof(eltype(a))) (from the ratio of element sizes)"))
804 end

809 """
810 resize!(a::CuVector, n::Integer)
811
812 Resize `a` to contain `n` elements. If `n` is smaller than the current collection length,
813 the first `n` elements will be retained. If `n` is larger, the new elements are not
814 guaranteed to be initialized.
815
816 Note that this operation is only supported on managed buffers, i.e., not on arrays that are
817 created by `unsafe_wrap` with `own=false`.
818 """
819 function Base.resize!(A::CuVector{T}, n::Integer) where T
820 # TODO: add additional space to allow for quicker resizing
821 maxsize = n * sizeof(T)
822 bufsize = if Base.isbitsunion(T)
823 # type tag array past the data
824 maxsize + n
825 else
826 maxsize
827 end
828
829 new_storage = context!(context(A)) do
830 buf = alloc(typeof(A.storage.buffer), bufsize)
831 ptr = convert(CuPtr{T}, buf)
832 m = min(length(A), n)
833 if m > 0
834 unsafe_copyto!(ptr, pointer(A), m)
835 end
836 ArrayStorage(buf, 1)
837 end
838
839 unsafe_free!(A)
840 A.storage = new_storage
841 A.dims = (n,)
842 A.maxsize = maxsize
843 A.offset = 0
844
845 A
