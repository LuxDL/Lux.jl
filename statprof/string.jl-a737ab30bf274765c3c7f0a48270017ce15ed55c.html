37 String(v::AbstractVector{UInt8})
38
39 Create a new `String` object from a byte vector `v` containing UTF-8 encoded
40 characters. If `v` is `Vector{UInt8}` it will be truncated to zero length and
41 future modification of `v` cannot affect the contents of the resulting string.
42 To avoid truncation of `Vector{UInt8}` data, use `String(copy(v))`; for other
43 `AbstractVector` types, `String(v)` already makes a copy.
44
45 When possible, the memory of `v` will be used without copying when the `String`
46 object is created. This is guaranteed to be the case for byte vectors returned
47 by [`take!`](@ref) on a writable [`IOBuffer`](@ref) and by calls to
48 [`read(io, nb)`](@ref). This allows zero-copy conversion of I/O data to strings.
49 In other cases, `Vector{UInt8}` data may be copied, but `v` is truncated anyway

334 function repeat(c::Char, r::Integer)
335 r == 0 && return ""
336 r < 0 && throw(ArgumentError("can't repeat a character $r times"))
337 u = bswap(reinterpret(UInt32, c))
338 n = 4 - (leading_zeros(u | 0xff) >> 3)
339 s = _string_n(n*r)
340 p = pointer(s)
341 GC.@preserve s if n == 1
342 ccall(:memset, Ptr{Cvoid}, (Ptr{UInt8}, Cint, Csize_t), p, u % UInt8, r)
343 elseif n == 2
344 p16 = reinterpret(Ptr{UInt16}, p)
345 for i = 1:r
346 unsafe_store!(p16, u % UInt16, i)
347 end
348 elseif n == 3
349 b1 = (u >> 0) % UInt8
350 b2 = (u >> 8) % UInt8
351 b3 = (u >> 16) % UInt8
352 for i = 0:r-1
353 unsafe_store!(p, b1, 3i + 1)
354 unsafe_store!(p, b2, 3i + 2)
355 unsafe_store!(p, b3, 3i + 3)
356 end
357 elseif n == 4
358 p32 = reinterpret(Ptr{UInt32}, p)
359 for i = 1:r
360 unsafe_store!(p32, u, i)
361 end
362 end
