244 # Check if two lattice elements are partial order equivalent. This is basically
245 # `a ⊑ b && b ⊑ a` but with extra performance optimizations.
246 function is_lattice_equal(@nospecialize(a), @nospecialize(b))

250 length(a.fields) == length(b.fields) || return false
