152 function strcat(x::String, y::String)
153 out = ccall(:jl_alloc_string, Ref{String}, (Csize_t,), Core.sizeof(x) + Core.sizeof(y))
154 GC.@preserve x y out begin
155 out_ptr = unsafe_convert(Ptr{UInt8}, out)
156 unsafe_copyto!(out_ptr, unsafe_convert(Ptr{UInt8}, x), Core.sizeof(x))
157 unsafe_copyto!(out_ptr + Core.sizeof(x), unsafe_convert(Ptr{UInt8}, y), Core.sizeof(y))
158 end
159 return out
160 end
