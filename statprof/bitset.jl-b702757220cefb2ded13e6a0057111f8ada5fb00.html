81 # An internal function for setting the inclusion bit for a given integer
82 @inline function _setint!(s::BitSet, idx::Int, b::Bool)
83 cidx = _div64(idx)
84 len = length(s.bits)
85 diff = cidx - s.offset
86 if diff >= len
87 b || return s # setting a bit to zero outside the set's bits is a no-op
88
89 # we put the following test within one of the two branches,
90 # with the NO_OFFSET trick, to avoid having to perform it at
91 # each and every call to _setint!
92 if s.offset == NO_OFFSET # initialize the offset
93 # we assume isempty(s.bits)
94 s.offset = cidx
95 diff = 0
96 end
97 _growend0!(s.bits, diff - len + 1)
98 elseif diff < 0
99 b || return s
100 _growbeg0!(s.bits, -diff)
101 s.offset += diff
102 diff = 0
103 end
104 _unsafe_bitsetindex!(s.bits, b, diff+1, _mod64(idx))
105 s
106 end

126 function union!(s::BitSet, r::AbstractUnitRange{<:Integer})
127 isempty(r) && return s
128 a, b = _check_bitset_bounds(first(r)), _check_bitset_bounds(last(r))
129 cidxa = _div64(a)
130 cidxb = _div64(b)
131 if s.offset == NO_OFFSET
132 s.offset = cidxa
133 end
134 len = length(s.bits)
135 diffa = cidxa - s.offset
136 diffb = cidxb - s.offset
137
138 # grow s.bits as necessary
139 if diffb >= len
140 _growend!(s.bits, diffb - len + 1)
141 # we set only some values to CHK0, those which will not be
142 # fully overwritten (i.e. only or'ed with `|`)
143 s.bits[end] = CHK0 # end == diffb + 1
144 if diffa >= len
145 s.bits[diffa + 1] = CHK0
146 end
147 end
148 if diffa < 0
149 _growbeg!(s.bits, -diffa)
150 s.bits[1] = CHK0
151 if diffb < 0
152 s.bits[diffb - diffa + 1] = CHK0
153 end
154 s.offset = cidxa # s.offset += diffa
155 diffb -= diffa
156 diffa = 0
157 end
158
159 # update s.bits
160 i = _mod64(a)
161 j = _mod64(b)
162 @inbounds if diffa == diffb
163 s.bits[diffa + 1] |= (((~CHK0) >> i) << (i+63-j)) >> (63-j)
164 else
165 s.bits[diffa + 1] |= ((~CHK0) >> i) << i
166 s.bits[diffb + 1] |= (~CHK0 << (63-j)) >> (63-j)
167 for n = diffa+1:diffb-1
168 s.bits[n+1] = ~CHK0
169 end
170 end
171 s
172 end

174 function _matched_map!(f, s1::BitSet, s2::BitSet)
175 left_false_is_false = f(false, false) == f(false, true) == false
176 right_false_is_false = f(false, false) == f(true, false) == false
177
178 # we must first handle the NO_OFFSET case; we could test for
179 # isempty(s1) but it can be costly, so the user has to call
180 # empty!(s1) herself before-hand to re-initialize to NO_OFFSET
181 if s1.offset == NO_OFFSET
182 return left_false_is_false ? s1 : copy!(s1, s2)
183 elseif s2.offset == NO_OFFSET
184 return right_false_is_false ? empty!(s1) : s1
185 end
186 s1.offset = _matched_map!(f, s1.bits, s1.offset, s2.bits, s2.offset,
187 left_false_is_false, right_false_is_false)
188 s1
189 end

191 # An internal function that takes a pure function `f` and maps across two BitArrays
192 # allowing the lengths and offsets to be different and altering b1 with the result
193 # WARNING: the assumptions written in the else clauses must hold
194 function _matched_map!(f, a1::Bits, b1::Int, a2::Bits, b2::Int,
195 left_false_is_false::Bool, right_false_is_false::Bool)
196 l1, l2 = length(a1), length(a2)
197 bdiff = b2 - b1
198 e1, e2 = l1+b1, l2+b2
199 ediff = e2 - e1
200
201 # map! over the common indices
202 @inbounds for i = max(1, 1+bdiff):min(l1, l2+bdiff)
203 a1[i] = f(a1[i], a2[i-bdiff])
204 end
205
206 if ediff > 0
207 if left_false_is_false
208 # We don't need to worry about the trailing bits they're all false
209 else # @assert f(false, x) == x
210 _growend!(a1, ediff)
211 # if a1 and a2 are not overlapping, we infer implied "false" values from a2
212 for outer l1 = l1+1:bdiff
213 @inbounds a1[l1] = CHK0
214 end
215 # update ediff in case l1 was updated
216 ediff = e2 - l1 - b1
217 # copy actual chunks from a2
218 unsafe_copyto!(a1, l1+1, a2, l2+1-ediff, ediff)
219 l1 = length(a1)
220 end
221 elseif ediff < 0
222 if right_false_is_false
223 # We don't need to worry about the trailing bits they're all false
224 _deleteend!(a1, min(l1, -ediff))
225 # no need to update l1, as if bdiff > 0 (case below), then bdiff will
226 # be smaller anyway than an updated l1
227 else # @assert f(x, false) == x
228 # We don't need to worry about the trailing bits they already have the
229 # correct value
230 end
231 end
232
233 if bdiff < 0
234 if left_false_is_false
235 # We don't need to worry about the leading bits they're all false
236 else # @assert f(false, x) == x
237 _growbeg!(a1, -bdiff)
238 # if a1 and a2 are not overlapping, we infer implied "false" values from a2
239 for i = l2+1:-bdiff
240 @inbounds a1[i] = CHK0
241 end
242 b1 += bdiff # updated return value
243
244 # copy actual chunks from a2
245 unsafe_copyto!(a1, 1, a2, 1, min(-bdiff, l2))
246 end
247 elseif bdiff > 0
248 if right_false_is_false
249 # We don't need to worry about the trailing bits they're all false
250 _deletebeg!(a1, min(l1, bdiff))
251 b1 += bdiff
252 else # @assert f(x, false) == x
253 # We don't need to worry about the trailing bits they already have the
254 # correct value
255 end
256 end
257 b1 # the new offset
258 end

385 function ==(s1::BitSet, s2::BitSet)
386 # Swap so s1 has always the smallest offset
387 if s1.offset > s2.offset
388 s1, s2 = s2, s1
389 end
390 a1 = s1.bits
391 a2 = s2.bits
392 b1, b2 = s1.offset, s2.offset
393 l1, l2 = length(a1), length(a2)
394 e1 = l1+b1
395 overlap0 = max(0, e1 - b2)
396 included = overlap0 >= l2 # whether a2's indices are included in a1's
397 overlap = included ? l2 : overlap0
398
399 # Ensure non-overlap chunks are zero (unlikely)
400 _check0(a1, 1, l1-overlap0) || return false
401 if included
402 _check0(a1, b2-b1+l2+1, l1) || return false
403 else
404 _check0(a2, 1+overlap, l2) || return false
405 end
406
407 # compare overlap values
408 if overlap > 0
409 t1 = @_gc_preserve_begin a1
410 t2 = @_gc_preserve_begin a2
411 _memcmp(pointer(a1, b2-b1+1), pointer(a2), overlap<<3) == 0 || return false
412 @_gc_preserve_end t2
413 @_gc_preserve_end t1
414 end
415
416 return true
417 end
