539 function batch_inline!(todo::Vector{Pair{Int, Any}}, ir::IRCode, linetable::Vector{LineInfoNode}, propagate_inbounds::Bool)
540 # Compute the new CFG first (modulo statement ranges, which will be computed below)
541 state = CFGInliningState(ir)
542 for (idx, item) in todo
543 if isa(item, UnionSplit)
544 cfg_inline_unionsplit!(ir, idx, item::UnionSplit, state)
545 else
546 item = item::InliningTodo
547 spec = item.spec::ResolvedInliningSpec
548 # A linear inline does not modify the CFG
549 spec.linear_inline_eligible && continue
550 cfg_inline_item!(ir, idx, spec, state, false)
551 end
552 end
553 finish_cfg_inline!(state)
554
555 boundscheck = inbounds_option()
556 if boundscheck === :default && propagate_inbounds
557 boundscheck = :propagate
558 end
559
560 let compact = IncrementalCompact(ir, false)
561 compact.result_bbs = state.new_cfg_blocks
562 # This needs to be a minimum and is more of a size hint
563 nn = 0
564 for (_, item) in todo
565 if isa(item, InliningTodo)
566 spec = item.spec::ResolvedInliningSpec
567 nn += (length(spec.ir.stmts) + length(spec.ir.new_nodes))
568 end
569 end
570 nnewnodes = length(compact.result) + nn
571 resize!(compact, nnewnodes)
572 (inline_idx, item) = popfirst!(todo)
573 for ((old_idx, idx), stmt) in compact
574 if old_idx == inline_idx
575 argexprs = copy(stmt.args)
576 refinish = false
577 if compact.result_idx == first(compact.result_bbs[compact.active_result_bb].stmts)
578 compact.active_result_bb -= 1
579 refinish = true
580 end
581 # It is possible for GlobalRefs and Exprs to be in argument position
582 # at this point in the IR, though in that case they are required
583 # to be effect-free. However, we must still move them out of argument
584 # position, since `Argument` is allowed in PhiNodes, but `GlobalRef`
585 # and `Expr` are not, so a substitution could anger the verifier.
