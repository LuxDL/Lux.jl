150 # eliminate UnionAll vars that might be degenerate due to having identical bounds,
151 # or a concrete upper bound and appearing covariantly.
152 function subst_trivial_bounds(@nospecialize(atypes))
153 if !isa(atypes, UnionAll)
154 return atypes
155 end
156 v = atypes.var
157 if isconcretetype(v.ub) || v.lb === v.ub
158 subst = try
159 atypes{v.ub}
160 catch
161 # Note in rare cases a var bound might not be valid to substitute.
162 nothing
163 end
164 if subst !== nothing
165 return subst_trivial_bounds(subst)
166 end
167 end
168 return UnionAll(v, subst_trivial_bounds(atypes.body))
169 end

228 function argextype(@nospecialize(x), src, sptypes::Vector{Any}, slottypes::Vector{Any} = empty_slottypes)
229 if isa(x, Expr)
230 if x.head === :static_parameter
231 return sptypes[x.args[1]]
232 elseif x.head === :boundscheck
233 return Bool
234 elseif x.head === :copyast
235 return argextype(x.args[1], src, sptypes, slottypes)
236 end
237 @assert false "argextype only works on argument-position values"
238 elseif isa(x, SlotNumber)
239 return slottypes[(x::SlotNumber).id]
240 elseif isa(x, TypedSlot)
241 return (x::TypedSlot).typ
242 elseif isa(x, SSAValue)
243 return abstract_eval_ssavalue(x::SSAValue, src)
244 elseif isa(x, Argument)
245 return isa(src, IncrementalCompact) ? src.ir.argtypes[x.n] :
246 isa(src, IRCode) ? src.argtypes[x.n] :
247 slottypes[x.n]
248 elseif isa(x, QuoteNode)
249 return Const((x::QuoteNode).value)
250 elseif isa(x, GlobalRef)

252 elseif isa(x, PhiNode)

254 elseif isa(x, PiNode)

321 function find_throw_blocks(code::Vector{Any}, ir = RefValue{IRCode}())
322 stmts = BitSet()
323 n = length(code)
324 try_depth = 0
325 for i in n:-1:1
326 s = code[i]
327 if isa(s, Expr)
328 if s.head === :enter
329 try_depth -= 1
330 elseif s.head === :leave
331 try_depth += (s.args[1]::Int)
332 elseif s.head === :gotoifnot
333 tgt = s.args[2]::Int
334 if i+1 in stmts && tgt in stmts
335 push!(stmts, i)
336 end
337 elseif s.head === :return
338 elseif is_throw_call(s)
339 if try_depth == 0
340 push!(stmts, i)
341 end
342 elseif i+1 in stmts
343 push!(stmts, i)
344 end
345 elseif isa(s, ReturnNode)
346 # NOTE: it potentially makes sense to treat unreachable nodes
347 # (where !isdefined(s, :val)) as `throw` points, but that can cause
348 # worse codegen around the call site (issue #37558)
349 elseif isa(s, GotoNode)
350 tgt = s.label
351 if isassigned(ir)
352 tgt = first(ir[].cfg.blocks[tgt].stmts)
353 end
354 if tgt in stmts
355 push!(stmts, i)
356 end
357 elseif isa(s, GotoIfNot)
358 if i+1 in stmts
359 tgt = s.dest::Int
360 if isassigned(ir)
361 tgt = first(ir[].cfg.blocks[tgt].stmts)
362 end
363 if tgt in stmts
364 push!(stmts, i)
365 end
366 end
367 elseif i+1 in stmts
368 push!(stmts, i)
369 end
370 end
371 return stmts
372 end
