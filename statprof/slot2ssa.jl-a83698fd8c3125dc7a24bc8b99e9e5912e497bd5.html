232 # Run iterated dominance frontier
233 #
234 # The algorithm we have here essentially follows LLVM, which itself is a
235 # a cleaned up version of the linear-time algorithm described in
236 #
237 # A Linear Time Algorithm for Placing phi-Nodes (by Sreedhar and Gao)
238 #
239 # The algorithm here, is quite straightforward. Suppose we have a CFG:
240 #
241 # A -> B -> D -> F
242 # \-> C -------/
243 #
244 # and a corresponding dominator tree:
245 #
246 # A
247 # |- B - D
248 # |- C
249 # |- F
250 #
251 # Now, for every definition of our slot, we simply walk down the dominator
252 # tree and look for any edges that leave the sub-domtree rooted by our definition.
253 #
254 # E.g. in our example above, if we have a definition in `B`, we look at its successors,
255 # which is only `D`, which is dominated by `B` and hence doesn't need a phi node.
256 # Then we descend down the subtree rooted at `B` and end up in `D`. `D` has a successor
257 # `F`, which is not part of the current subtree, (i.e. not dominated by `B`), so it
258 # needs a phi node.
259 #
260 # Now, the key insight of that algorithm is that we have two defs, in blocks `A` and `B`,
261 # and `A` dominates `B`, then we do not need to recurse into `B`, because the set of
262 # potential backedges from a subtree rooted at `B` (to outside the subtree) is a strict
263 # subset of those backedges from a subtree rooted at `A` (out outside the subtree rooted
264 # at `A`). Note however that this does not work the other way. Thus, the algorithm
265

267 # This should be a priority queue, but TODO - sorted array for now
268 defs = liveness.def_bbs
269 pq = Tuple{Int, Int}[(defs[i], domtree.nodes[defs[i]].level) for i in 1:length(defs)]
270 sort!(pq, by=x->x[2])
271 phiblocks = Int[]
272 # This bitset makes sure we only add a phi node to a given block once.
273 1 (0 %)
1 (100 %) samples spent calling BitSet
processed = BitSet()
274 # This bitset implements the `key insight` mentioned above. In particular, it prevents
275 # us from visiting a subtree that we have already visited before.
276 visited = BitSet()
277 while !isempty(pq)
278 # We pop from the end of the array - i.e. the element with the highest level.
279 node, level = pop!(pq)
280 worklist = Int[]
281 push!(worklist, node)
282 while !isempty(worklist)
283 active = pop!(worklist)
284 for succ in cfg.blocks[active].succs
285 # Check whether the current root (`node`) dominates succ.
286 # We are guaranteed that `node` dominates `active`, since
287 # we've arrived at `active` by following dominator tree edges.
288 # If `succ`'s level is less than or equal to that of `node`,
289 # it cannot possibly be dominated by `node`. On the other hand,
290 # since at this point we know that there is an edge from `node`'s
291 # subtree to `succ`, we know that if succ's level is greater than
292 # that of `node`, it must be dominated by `node`.
293 succ_level = domtree.nodes[succ].level
294 succ_level > level && continue
295 # We don't dominate succ. We need to place a phinode,
296 # unless liveness said otherwise.
297 succ in processed && continue
298 push!(processed, succ)
299 if !(succ in liveness.live_in_bbs)
300 continue
301 end
302 push!(phiblocks, succ)
303 # Basically: Consider the phi node we just added another
304 # def of this value. N.B.: This needs to retain the invariant that it
305 # is processed before any of its parents in the dom tree. This is guaranteed,
306 # because succ_level <= level, which is the greatest level we have currently
307 # processed. Thus, we have not yet processed any subtrees of level < succ_level.
308 if !(succ in defs)
309 push!(pq, (succ, succ_level))
310 sort!(pq, by=x->x[2])
311 end
312 end
313 # Recurse down the current subtree
314 for child in domtree.nodes[active].children
315 child in visited && continue
316 push!(visited, child)
317 push!(worklist, child)
318 end
319 end

516 function compute_live_ins(cfg::CFG, defuse #=::Union{SlotInfo,SSADefUse}=#)
517 # We remove from `uses` any block where all uses are dominated
518 # by a def. This prevents insertion of dead phi nodes at the top
519 # of such a block if that block happens to be in a loop
520 ordered = Tuple{Int, Int, Bool}[(x, block_for_inst(cfg, x), true) for x in defuse.uses]
521 for x in defuse.defs
522 push!(ordered, (x, block_for_inst(cfg, x), false))
523 end
524 ordered = sort(ordered, by=x->x[1])
525 bb_defs = Int[]
526 bb_uses = Int[]
527 last_bb = last_def_bb = 0
528 for (_, bb, is_use) in ordered
529 if bb != last_bb && is_use
530 push!(bb_uses, bb)
531 end
532 last_bb = bb
533 if last_def_bb != bb && !is_use
534 push!(bb_defs, bb)
535 last_def_bb = bb
536 end
537 end
538 # To obtain live ins from bb_uses, recursively add predecessors
539 extra_liveins = BitSet()
540 worklist = Int[]
541 for bb in bb_uses
542 append!(worklist, filter(p->p != 0 && !(p in bb_defs), cfg.blocks[bb].preds))
543 end
544 while !isempty(worklist)
545 elem = pop!(worklist)
546 (elem in bb_uses || elem in extra_liveins) && continue
547 push!(extra_liveins, elem)
548 append!(worklist, filter(p->p != 0 && !(p in bb_defs), cfg.blocks[elem].preds))
549 end
550 append!(bb_uses, extra_liveins)
551 BlockLiveness(bb_defs, bb_uses)
552 end

594 exc_handlers[enter_block+1] = (enter_block, exc)
595 # TODO: Cut off here if the terminator is a leave corresponding to this enter
596 for block in dominated(domtree, enter_block+1)
597 exc_handlers[block] = (enter_block, exc)

646 if !(li in live.def_bbs)
647 push!(live.def_bbs, li)

651 1 (0 %)
1 (100 %) samples spent calling idf
phiblocks = idf(cfg, live, domtree)

655 ssa = NewSSAValue(insert_node!(ir,

668 SSAValue(-2)

675 @timeit "SSA Rename" while !isempty(worklist)

694 for (idx, slot) in Iterators.enumerate(phi_slots[item])

719 old_entry[:type] = new_typ

746 (isa(stmt, PhiNode) || (isexpr(stmt, :(=)) && isa(stmt.args[2], PhiNode))) && continue
747 if isa(stmt, NewvarNode)

766 else
767 code[idx] = nothing
768 incoming_vals[id] = undef_token

784 end
785 end
786 end

789 end

791 # Delete any instruction in unreachable blocks (except for terminators)
792 for bb in setdiff(BitSet(1:length(cfg.blocks)), visited)
793 for idx in cfg.blocks[bb].stmts
794 if isa(code[idx], Union{GotoNode, GotoIfNot, ReturnNode})
795 code[idx] = ReturnNode()
796 else
797 code[idx] = nothing

801 # Convert into IRCode form
802 nstmts = length(ir.stmts)
803 new_code = Vector{Any}(undef, nstmts)

806 # Detect statement positions for assignments and construct array
807 for (bb, idx) in bbidxiter(ir)
808 stmt = code[idx]
809 # Convert GotoNode/GotoIfNot/PhiNode to BB addressing
810 if isa(stmt, GotoNode)

812 elseif isa(stmt, GotoIfNot)
813 new_dest = block_for_inst(cfg, stmt.dest)
814 if new_dest == bb+1
815 # Drop this node - it's a noop
816 new_code[idx] = Expr(:call, GlobalRef(Core, :typeassert), stmt.cond, GlobalRef(Core, :Bool))
817 else
818 new_code[idx] = GotoIfNot(stmt.cond, new_dest)
819 end
820 elseif isexpr(stmt, :enter)
821 new_code[idx] = Expr(:enter, block_for_inst(cfg, stmt.args[1]))

823 elseif isexpr(stmt, :leave) || isexpr(stmt, :(=)) || isa(stmt, ReturnNode) ||
824 isexpr(stmt, :meta) || isa(stmt, NewvarNode)

826 else
827 ssavalmap[idx] = SSAValue(idx)
828 result_types[idx] = ci.ssavaluetypes[idx]
829 if isa(stmt, PhiNode)
830 edges = Int32[edge == 0 ? 0 : block_for_inst(cfg, Int(edge)) for edge in stmt.edges]
831 new_code[idx] = PhiNode(edges, stmt.values)
832 else
833 new_code[idx] = stmt

836 end

839 new_typ = Union{}

847 @assert !isa(typ, MaybeUndef)

850 end

852 end
853 node[:type] = new_typ
854 end
855 end

866 node[:type] = new_typ
