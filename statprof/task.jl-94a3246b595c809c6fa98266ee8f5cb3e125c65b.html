356 function sync_end(c::Channel{Any})
357 local c_ex
358 while isready(c)
359 r = take!(c)
360 if isa(r, Task)
361 _wait(r)
362 if istaskfailed(r)
363 if !@isdefined(c_ex)
364 c_ex = CompositeException()
365 end
366 push!(c_ex, TaskFailedException(r))
367 end
368 else
369 try
370 wait(r)
371 catch e
372 if !@isdefined(c_ex)
373 c_ex = CompositeException()
374 end
375 push!(c_ex, e)
376 end
377 end
378 end
379 close(c)
380 if @isdefined(c_ex)
381 throw(c_ex)
382 end
383 nothing

475 # Capture interpolated variables in $() and move them to let-block
476 function _lift_one_interp!(e)
477 letargs = Any[] # store the new gensymed arguments
478 _lift_one_interp_helper(e, false, letargs) # Start out _not_ in a quote context (false)
479 letargs
480 end
481 _lift_one_interp_helper(v, _, _) = v
482 function _lift_one_interp_helper(expr::Expr, in_quote_context, letargs)
483 if expr.head === :$
484 if in_quote_context # This $ is simply interpolating out of the quote
485 # Now, we're out of the quote, so any _further_ $ is ours.
486 in_quote_context = false
487 else
488 newarg = gensym()
489 push!(letargs, :($(esc(newarg)) = $(esc(expr.args[1]))))
490 return newarg # Don't recurse into the lifted $() exprs
491 end
492 elseif expr.head === :quote
493 in_quote_context = true # Don't try to lift $ directly out of quotes
494 elseif expr.head === :macrocall
495 return expr # Don't recur into macro calls, since some other macros use $
496 end
497 for (i,e) in enumerate(expr.args)
498 expr.args[i] = _lift_one_interp_helper(e, in_quote_context, letargs)
499 end
500 expr
501 end

514 # runtime system hook called when a task finishes
515 function task_done_hook(t::Task)
516 # `finish_task` sets `sigatomic` before entering this function
517 err = istaskfailed(t)
518 result = task_result(t)
519 handled = false
520
521 donenotify = t.donenotify
522 if isa(donenotify, ThreadSynchronizer)
523 lock(donenotify)
524 try
525 if !isempty(donenotify.waitq)
526 handled = true
527 notify(donenotify)
528 end
529 finally
530 unlock(donenotify)
531 end
532 end
533
534 if err && !handled && Threads.threadid() == 1
535 if isa(result, InterruptException) && isdefined(Base, :active_repl_backend) &&
536 active_repl_backend.backend_task._state === task_state_runnable && isempty(Workqueue) &&
537 active_repl_backend.in_eval
538 throwto(active_repl_backend.backend_task, result) # this terminates the task
539 end
540 end
541 # Clear sigatomic before waiting
542 sigatomic_end()
543 try
544 wait() # this will not return
545 catch e
546 # If an InterruptException happens while blocked in the event loop, try handing
547 # the exception to the REPL task since the current task is done.
548 # issue #19467
549 if Threads.threadid() == 1 &&
550 isa(e, InterruptException) && isdefined(Base, :active_repl_backend) &&
551 active_repl_backend.backend_task._state === task_state_runnable && isempty(Workqueue) &&
552 active_repl_backend.in_eval
553 throwto(active_repl_backend.backend_task, e)
554 else
555 rethrow()
556 end
557 end

627 function enq_work(t::Task)
628 (t._state === task_state_runnable && t.queue === nothing) || error("schedule: Task not runnable")
629 tid = Threads.threadid(t)
630 # Note there are three reasons a Task might be put into a sticky queue
631 # even if t.sticky == false:
632 # 1. The Task's stack is currently being used by the scheduler for a certain thread.
633 # 2. There is only 1 thread.
634 # 3. The multiq is full (can be fixed by making it growable).
635 if t.sticky || Threads.nthreads() == 1
636 if tid == 0
637 # Issue #41324
638 # t.sticky && tid == 0 is a task that needs to be co-scheduled with
639 # the parent task. If the parent (current_task) is not sticky we must
640 # set it to be sticky.
641 # XXX: Ideally we would be able to unset this
642 current_task().sticky = true
643 tid = Threads.threadid()
644 ccall(:jl_set_task_tid, Cvoid, (Any, Cint), t, tid-1)
645 end
646 push!(Workqueues[tid], t)
647 else
648 if ccall(:jl_enqueue_task, Cint, (Any,), t) != 0
649 # if multiq is full, give to a random thread (TODO fix)
650 if tid == 0
651 tid = mod(time_ns() % Int, Threads.nthreads()) + 1
652 ccall(:jl_set_task_tid, Cvoid, (Any, Cint), t, tid-1)
653 end
654 push!(Workqueues[tid], t)
655 else
656 tid = 0
657 end
658 end
659 ccall(:jl_wakeup_thread, Cvoid, (Int16,), (tid - 1) % Int16)
660 return t
661 end

666 schedule(t::Task, [val]; error=false)
667
668 Add a [`Task`](@ref) to the scheduler's queue. This causes the task to run constantly when the system
669 is otherwise idle, unless the task performs a blocking operation such as [`wait`](@ref).
670
671 If a second argument `val` is provided, it will be passed to the task (via the return value of
672 [`yieldto`](@ref)) when it runs again. If `error` is `true`, the value is raised as an exception in
673 the woken task.
674

676 ```jldoctest

678

690
691 julia> istaskdone(b)

693 ```

695 function schedule(t::Task, @nospecialize(arg); error=false)
696 # schedule a task to be (re)started with the given value or exception
697 t._state === task_state_runnable || Base.error("schedule: Task not runnable")
698 if error
699 t.queue === nothing || Base.list_deletefirst!(t.queue, t)
700 setfield!(t, :result, arg)
701 setfield!(t, :_isexception, true)
702 else
703 t.queue === nothing || Base.error("schedule: Task not runnable")
704 setfield!(t, :result, arg)
705 end
706 enq_work(t)
707 return t
708 end
