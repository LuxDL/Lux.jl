115 # return an upper-bound on type `a` with type `b` removed
116 # such that `return <: a` && `Union{return, b} == Union{a, b}`
117 function typesubtract(@nospecialize(a), @nospecialize(b), MAX_UNION_SPLITTING::Int)
118 if a <: b && isnotbrokensubtype(a, b)
119 return Bottom
120 end

125 elseif a isa DataType
126 ub = unwrap_unionall(b)
127 if ub isa DataType
128 if a.name === ub.name === Tuple.name &&
129 length(a.parameters) == length(ub.parameters)
130 if 1 < unionsplitcost(a.parameters) <= MAX_UNION_SPLITTING
131 ta = switchtupleunion(a)
132 return typesubtract(Union{ta...}, b, 0)
133 elseif b isa DataType
134 if !compatible_vatuple(a, b)
135 return a
136 end
137 # if exactly one element is not bottom after calling typesubtract
138 # then the result is all of the elements as normal except that one
139 notbottom = fill(false, length(a.parameters))
140 for i = 1:length(notbottom)
141 ap = unwrapva(a.parameters[i])
142 bp = unwrapva(b.parameters[i])
143 notbottom[i] = !(ap <: bp && isnotbrokensubtype(ap, bp))

147 ta = collect(a.parameters)
148 ap = a.parameters[i]

191 # Gives a cost function over the effort to switch a tuple-union representation
192 # as a cartesian product, relative to the size of the original representation.
193 # Thus, we count the longest element as being roughly invariant to being inside
194 # or outside of the Tuple/Union nesting, though somewhat more expensive to be
195 # outside than inside because the representation is larger (because and it
196 # informs the callee whether any splitting is possible).
197 function unionsplitcost(atypes::Union{SimpleVector,Vector{Any}})
198 nu = 1
199 max = 2
200 for ti in atypes
201 if isa(ti, Union)
202 nti = unionlen(ti)
203 if nti > max
204 max, nti = nti, max
205 end
206 nu, ovf = Core.Intrinsics.checked_smul_int(nu, nti)
207 ovf && return typemax(Int)
208 end
209 end
210 return nu
211 end
